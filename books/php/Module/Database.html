# PHP 数据库

这里只讲 PHP 连接 MySQL 数据库。PHP 用于操作 MySQL 数据库的扩展包括 PDO、 Mysql 和 Mysqli，其中，Mysqli 是 MySQL 增强版扩展。

此类扩展是需要安装的，安装参考 mysqli 的安装、PDO 的安装

  • mysqli (面向对象、面向过程)
  • mysql (面向过程)
  • PDO (PHP Data Objects)

官方文档: mysqli、mysql

# 连接和关闭数据库

使用 mysqli 扩展，面向对象风格

// 创建连接
$servername = 'localhost';
$username = 'my_user';
$password = 'my_password';
$database = 'my_db';
$conn = new mysqli($servername, $username, $password, $database);
 
// 检测连接
if ($conn->connect_error) {
  die("连接失败: " . $conn->connect_error);
} 
echo "连接成功";

使用 mysqli 扩展，面向过程风格

$con = mysqli_connect("localhost","peter","abc123");
if (!$con) {
  die('Connect Error (' . mysqli_connect_errno() . ') '
    . mysqli_connect_error());
}

echo 'Success... ' . mysqli_get_host_info($link) . "\n";

mysqli_close($link);

使用 mysql 扩展

$con = mysql_connect("localhost","peter","abc123");
if (!$con)
{
  die('Could not connect: ' . mysql_error());
}

mysql_close($con);

使用 PDO 扩展

$servername = "localhost";
$username = "username";
$password = "password";
 
try {
  $conn = new PDO("mysql:host=$servername;dbname=myDB", $username, $password);
  echo "连接成功"; 
}
catch(PDOException $e)
{
  echo $e->getMessage();
}

# 指定数据库

mysqli 可在连接数据库的时候直接指定数据库名称，mysql 需要在连接后指定数据库名称:

使用 mysqli，连接的时候指定数据库名称

$con = mysqli_connect("localhost","peter","abc123","my_db");

使用 mysqli，之后指定连接的数据库名称

$con = mysqli_connect("localhost","peter","abc123");
mysqli_select_db("my_db", $con);

使用 mysqli，面向对象的写法

$mysqli = new mysqli("localhost", "peter", "abc123", "test");
$mysqli->select_db("my_db");

使用 mysql

$con = mysql_connect("localhost","peter","abc123");
mysql_select_db("my_db", $con);

# 执行查询语句

使用 mysql_query 执行查询语句，比如

普通查询

if (!mysql_query("CREATE DATABASE my_db", $con)) {
  die('Invalid query: ' . mysql_error());
}

将格式化字串传入查询条件

$query = sprintf("SELECT firstname, lastname, address, age FROM friends 
  WHERE firstname='%s' AND lastname='%s'",
  mysql_real_escape_string($firstname),
  mysql_real_escape_string($lastname));
$result = mysql_query($query, $con);

使用 mysqli，面向对象的查询方式

$mysqli = new mysqli("localhost", "my_user", "my_password", "world");

if ($mysqli->query("CREATE TEMPORARY TABLE myCity LIKE City") === TRUE) {
  printf("Table myCity successfully created.\n");
}

使用 mysqli，面向过程的查询方式

$sql = "CREATE DATABASE myDB";
if (mysqli_query($conn, $sql)) {
  echo "数据库创建成功";
} else {
  echo "Error creating database: " . mysqli_error($conn);
}

使用 PDO

try {
  $conn = new PDO("mysql:host=localhost;dbname=myDB", 'my_user', 'my_password'); 

  // 设置 PDO 错误模式为异常 
  $conn->setAttribute(PDO::ATTR_ERRMODE, PDO::ERRMODE_EXCEPTION); 
  $sql = "CREATE DATABASE myDBPDO"; 

  // 使用 exec() ，没有结果返回
  $conn->exec($sql); 

  echo "数据库创建成功<br>";
}
catch(PDOException $e)
{
  echo $sql . "<br>" . $e->getMessage();
}

$conn = null;

# 查询结果处理

# mysql_fetch_object

mysql_fetch_object() 从结果集中取得一行作为对象。

mysql_connect("hostname", "user", "password");
mysql_select_db("mydb");
$result = mysql_query("select * from mytable");
while ($row = mysql_fetch_object($result)) {
  echo $row->user_id;
  echo $row->fullname;
}
mysql_free_result($result);

# mysql_fetch_row

mysql_fetch_row() 从结果集中取得一行作为枚举数组。

mysql_connect("hostname", "user", "password");
mysql_select_db("mydb");
$result = mysql_query("select * from mytable");
while ($row = mysql_fetch_row($result)) {
  echo $row[0];
  echo $row[1];
}
mysql_free_result($result);

# mysqli_fetch_assoc

mysqli_fetch_assoc() 从结果集中取得一行作为关联数组。

$conn = mysqli_connect("localhost", "mysql_user", "mysql_password", "mydb");

$result = mysqli_query($conn, "select * from mytable");

if (mysqli_num_rows($result) > 0) {
  // 输出数据
  while($row = mysqli_fetch_assoc($result)) {
    echo $row["id"];
    echo $row["fullname"];
  }
} else {
  echo "0 结果";
}

# mysqli_fetch_fields

mysql_fetch_fields() 从结果集中取得列信息并作为对象返回。

$link = mysqli_connect("127.0.0.1", "my_user", "my_password", "my_db");

if ($result = mysqli_query($link, $query)) {
    $finfo = mysqli_fetch_fields($result);
    foreach ($finfo as $val) {
      printf("Name:      %s\n",   $val->name);
      printf("Table:     %s\n",   $val->table);
      printf("Max. Len:  %d\n",   $val->max_length);
      printf("Length:    %d\n",   $val->length);
      printf("charsetnr: %d\n",   $val->charsetnr);
      printf("Flags:     %d\n",   $val->flags);
      printf("Type:      %d\n\n", $val->type);
    }
    mysqli_free_result($result);
}

# mysqli_fetch_array

mysql_fetch_array() 从结果集中取得一行作为关联数组，或数字数组，或二者兼有。

mysqli_connect("localhost", "my_user", "my_password", "world");

$query = "SELECT Name, CountryCode FROM City ORDER by ID LIMIT 3";
$result = mysqli_query($link, $query);

/* numeric array */
while ($row = mysqli_fetch_array($result, MYSQL_NUM)) {
  printf ("ID: %s  Name: %s", $row[0], $row[1]);
}

/* associative array */
while ($row = mysqli_fetch_array($result, MYSQLI_ASSOC)) {
  printf ("ID: %s  Name: %s", $row["ID"], $row["Name"]);
}

/* associative and numeric array */
while ($row = mysqli_fetch_array($result, MYSQLI_BOTH)) {
  printf ("ID: %s  Name: %s", $row[0], $row["Name"]);
}

mysqli_free_result($result);
