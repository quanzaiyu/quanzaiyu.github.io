# 音频

# react-native-audio-toolkit

react-native-audio-toolkit

# 安装

$ yarn add react-native-audio-toolkit
$ react-native link react-native-audio-toolkit

也可手动配置:

在 android/settings.gradle 中:

include ':react-native-audio-toolkit'
project(':react-native-audio-toolkit').projectDir = new File(rootProject.projectDir, '../node_modules/react-native-audio-toolkit/android/lib')

在 android/app/build.gradle 中

dependencies {
  implementation project(':react-native-audio-toolkit')
  compile project(':react-native-audio-toolkit')
}

在 MainActivity.java (react-native < 0.29) 或 MainApplication.java (react-native >= 0.29) 中:

//...
import com.futurice.rctaudiotoolkit.AudioPackage;
//...
@Override
protected List<ReactPackage> getPackages() {
  return Arrays.<ReactPackage>asList(
    new MainReactPackage(),
    new AudioPackage()
  );
}

AndroidManifest.xml 加入相关权限, RECORD_AUDIO 为录音权限, WRITE_EXTERNAL_STORAGE 为写入数据权限

<uses-permission android:name="android.permission.RECORD_AUDIO" />
<uses-permission android:name="android.permission.WRITE_EXTERNAL_STORAGE" />

# 使用

# 播放音频相关API

示例

import React from 'react';
import {
    View,
    Button
} from 'react-native';

import {
    Player
} from 'react-native-audio-toolkit';

let filename = 'https://zys201811.oss-cn-shenzhen.aliyuncs.com/media/0.mp3';

export default class AppContainer extends React.Component {
    constructor() {
        super();
        this.player = new Player(filename, {
            autoDestroy: false
        })
        this.player.prepare(function () {
            console.log("prepared")
        })
    }

    _play () {
        this.player.looping = true;
        this.player.volume = 0.9;
        this.player.play(() => {console.log("play invoke")})
    }

    _pause () {
        this.player.pause(() => {console.log("pause invoke")})
    }

    _stop () {
        this.player.stop(() => {console.log("stop invoke")})
    }

    _seek () {
        this.player.seek(30000, () => {console.log("seek invoke")})
    }

    render() {
        return (
            <View>
                <Button title={'开始播放'} onPress={() => this._play()}></Button>
                <Button title={'暂停播放'} onPress={() => this._pause()}></Button>
                <Button title={'停止播放'} onPress={() => this._stop()}></Button>
                <Button title={'跳转到30s处'} onPress={() => this._seek()}></Button>
            </View>
        );
    }
}

创建播放器

  • new Player(String path, Object ?playbackOptions) 创建音乐播放器
    • path 可以为网络地址, 也可为本地地址
    • 支持常见的音频和视频格式, 视频只抽取其音轨

事件处理

  • prepare(Function callback) 准备开始播放的回调

实例方法

  • play(Function ?callback) 开始播放
  • pause(Function ?callback) 暂停播放
  • playPause(Function ?callback) 切换播放与暂停状态
  • stop(Function ?callback) 停止播放
  • destroy(Function ?callback) 销毁播放器
  • seek(Number position, Function ?callback) 跳转到指定位置(ms)

实例属性

  • looping 是否重复
  • volume 音量大小
  • duration 时长(read only)
  • currentTime 当前时间
  • state 播放状态
var MediaStates = {
      DESTROYED: -2,
      ERROR: -1,
      IDLE: 0,
      PREPARING: 1,
      PREPARED: 2,
      SEEKING: 3,
      PLAYING: 4,   // only for Player
      RECORDING: 4, // only for Recorder
      PAUSED: 5
    };

Helpers for states - Boolean (read only)

Player.canPlay      true if player can begin playback
Player.canStop      true if player can stop playback
Player.canPrepare   true if player can prepare for playback
Player.isPlaying    true if player is playing
Player.isStopped    true if player is stopped
Player.isPaused     true if player is paused
Player.isPrepared   true if player is prepared

# 录音相关API

import React from 'react';
import {
    View,
    Button
} from 'react-native';

import {
    Recorder
} from 'react-native-audio-toolkit';

let filename = 'test.mp4';

export default class AppContainer extends React.Component {
    constructor() {
        super();
        this.recorder = new Recorder(filename, {
            bitrate: 256000,
            channels: 2,
            sampleRate: 44100,
            quality: 'max'
            //format: 'ac3', // autodetected
            //encoder: 'aac', // autodetected
        });
    }

    _record () {
        this.recorder.record(() => {console.log("record invoke")})
    }

    _stop () {
        this.recorder.stop(() => {console.log("stop invoke")})
    }

    render() {
        return (
            <View>
                <Button title={'开始录音'} onPress={() => this._record()}></Button>
                <Button title={'停止录音'} onPress={() => this._stop()}></Button>
            </View>
        );
    }
}
