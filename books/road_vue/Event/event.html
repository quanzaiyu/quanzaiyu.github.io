事件对象

可以在调用方法的时候传入一个 $event，这就是事件对象。

import Vue from 'vue'

let template = `<h1 @click='func("welcome", $event)'>{{text}}</h1>`

/* eslint-disable no-new */
new Vue({
  el: '#app',
  template,
  data: {
    text: 'Hello'
  },
  methods: {
    func (data = 'Hello', event) {
      this.text = data
      console.log(event)
    }
  }
})
