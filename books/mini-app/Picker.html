# Picker

官方文档中并没有把 Picker 说得非常详细, 特别是对象数组形式的单列选择和多列选择, 这里大概补充一下, 以备以后方便使用。

# 对象形式的多列选择

index.wxml

<view class="container">
  <view class="section">
    <picker
      mode="multiSelector"
      bindchange="bindMultiPickerChange"
      bindcolumnchange="bindMultiPickerColumnChange"
      range-key="name"
      range="{{showList}}"
    >
      <view class="picker">
        当前选择：{{choosed.first.name}} - {{choosed.second.name}}
      </view>
    </picker>
  </view>
</view>

index.js

  data: {
    list: { "code": 0, "msg": "success", "data": { "list": [{ "id": "1", "pid": "0", "name": "互联网IT行业", "_child": [{ "id": "10", "pid": "1", "name": "Java开发" }, { "id": "11", "pid": "1", "name": "UI设计师" }, { "id": "12", "pid": "1", "name": "Web前端" }, { "id": "13", "pid": "1", "name": "PHP" }, { "id": "14", "pid": "1", "name": "美工" }] }, { "id": "3", "pid": "0", "name": "房地产/建筑", "_child": [{ "id": "26", "pid": "3", "name": "房产经纪人" }, { "id": "27", "pid": "3", "name": "置业顾问" }, { "id": "28", "pid": "3", "name": "房产店长" }, { "id": "29", "pid": "3", "name": "房地产开发/策划" }, { "id": "30", "pid": "3", "name": "房产客服" }, { "id": "31", "pid": "3", "name": "房产内勤" }, { "id": "32", "pid": "3", "name": "房产评估" }] }, { "id": "5", "pid": "0", "name": "市场/销售", "_child": [{ "id": "33", "pid": "5", "name": "销售经理" }, { "id": "34", "pid": "5", "name": "销售总监" }, { "id": "35", "pid": "5", "name": "销售助理" }, { "id": "36", "pid": "5", "name": "销售代表" }, { "id": "37", "pid": "5", "name": "电话销售" }, { "id": "38", "pid": "5", "name": "汽车销售" }, { "id": "39", "pid": "5", "name": "网络销售" }, { "id": "40", "pid": "5", "name": "客户代表" }] }] } },
    showList: [],
    choosed: {},
    first: {},
    second: {}
  },
  onLoad: function () {
    let data = this.data.list.data.list
    let showList = []
    showList[0] = data
    showList[1] = data[0]._child
    this.setData({
      showList
    })
  },
  bindMultiPickerChange (e) {
    console.log(e)
  },
  bindMultiPickerColumnChange(e) {
    console.log(e.detail)
    let first = {}
    let second = {}
    if (e.detail.column === 0) {
      let showList = []
      showList[0] = this.data.showList[0]
      showList[1] = this.data.showList[0][e.detail.value]._child
      this.setData({
        first: {
          id: this.data.showList[0][e.detail.value].id,
          name: this.data.showList[0][e.detail.value].name
        },
        showList
      })
    }
    if (e.detail.column === 1) {
      this.setData({
        second: {
          id: this.data.showList[1][e.detail.value].id,

      choosed: {

        second: this.data.second

})

这里使用 data.list 模拟从接口获取到的数据, showList 为需要在 picker 中显示的数据, 将 picker 的 mode 声明为 multiSelector, 即可接收一个多列显示的 picker.

此处使用 choosed 存储选中的项, 并在 bindcolumnchange 事件中监听 picker 的列改变, 改变时就存储新的数据.

附上 mode=multiSelector 的参数:

属性名 类型 默认值 说明 最低版本
range 二维Array / 二维Object Array [] mode为 selector 或 multiSelector 时，range 有效。二维数组，长度表示多少列，数组的每项表示每列的数据，如[["a","b"], ["c","d"]]
range-key String 当 range 是一个 二维Object Array 时，通过 range-key 来指定 Object 中 key 的值作为选择器显示内容
value Array [] value 每一项的值表示选择了 range 对应项中的第几个（下标从 0 开始）
bindchange EventHandle value 改变时触发 change 事件，event.detail = {value: value}
bindcolumnchange EventHandle 某一列的值改变时触发 columnchange 事件，event.detail = {column: column, value: value}，column 的值表示改变了第几列（下标从0开始），value 的值表示变更值的下标
bindcancel EventHandle 取消选择时触发 1.9.90
disabled Boolean false 是否禁用
