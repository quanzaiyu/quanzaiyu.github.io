# array_column

array_column(array,column_key,index_key) 返回输入数组中某个单一列的值。

不指定键名:

$a = array(
  array(
    'id' => 5698,
    'first_name' => 'Peter',
    'last_name' => 'Griffin',
  ),
  array(
    'id' => 4767,
    'first_name' => 'Ben',
    'last_name' => 'Smith',
  ),
  array(
    'id' => 3809,
    'first_name' => 'Joe',
    'last_name' => 'Doe',
  )
);

$last_names = array_column($a, 'last_name');
print_r($last_names);

    'last_name' => 'Smith',

  array(
    'id' => 3809,
    'first_name' => 'Joe',
    'last_name' => 'Doe',
  )

$last_names = array_column($a, 'last_name', 'id');
