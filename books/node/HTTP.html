# 网络编程 HTTP

Node.js 提供了 http 模块，http 模块主要用于搭建 HTTP 服务端和客户端，使用 HTTP 服务器或客户端功能必须调用 http 模块。

# HTTP服务开发步骤

  1. 引入 HTTP 模块
const http = require('http')
  1. 使用 createServer 创建HTTP服务器
const server = http.createServer((req, res) => {

});
  1. 使用 listen 监听端口
server.listen(8080)

# request 和 response

createServer 需要接受一个回调函数，回调函数带两个参数: request 和 response

  • request对象封装了HTTP请求，我们调用request对象的属性和方法就可以拿到所有HTTP请求的信息；
  • response对象封装了HTTP响应，我们操作response对象的方法，就可以把HTTP响应返回给浏览器。

# request.url

使用 request.url 实现简单的路由:

switch (req.url) {
    case '/index.html':
    res.write('Hello world')
    break;
  case '/favicon.ico':
    res.write('This is a icon')
    break;
  default:
    res.write('404')
    break;
}

在浏览器中输入 http://localhost:8080/index.html 和 http://localhost:8080/favicon.ico 即可看到效果。

# response

// 向浏览器返回数据
res.write('Hello world')
// 设置文件头，将HTTP响应200写入response, 同时设置Content-Type: text/html:
res.writeHead(200, {'Content-Type': 'text/html'});
// 响应结束
res.end()
// end也可以指定参数，向浏览器输出
response.end('<h1>Hello world!</h1>');

# 完整代码

const http = require('http')

const server = http.createServer((req, res) => {
  console.log(req.url)
  res.writeHead(200, {'Content-Type': 'text/html'});
  switch (req.url) {
    case '/index.html':
      res.write('Hello world')
      break;
    case '/favicon.ico':
      res.write('This is a icon')
      break;
    default:
      res.write('404')
      break;
  }
  res.end()
});

server.listen(8080)

console.log('Server is running at http://127.0.0.1:8080/');

# 文件服务器

我们在编写服务器代码的时候，不可能将所有的请求路径都判断一遍，然后针对不同的请求返回不同的数据。此时，我们就需要将文件存储到磁盘，通过文件系统操作返回对应的数据。

先看看代码

const http = require('http')
const fs = require('fs')

const server = http.createServer((req, res) => {
  var file_name = './www/' + req.url
  fs.readFile(file_name, (err, data) => {
    if (err) {
      res.write('404')
    } else {
      res.write(data)
    }
    res.end()
  })
});

server.listen(8080)

在 www 目录下新建一个 index.html，写入:

<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <meta http-equiv="X-UA-Compatible" content="ie=edge">
  <title>Document</title>
  <style>
    .box {
      width: 200px;
      height: 200px;
      background-color: #f00;
    }
  </style>
</head>
<body>
  <div class="box"></div>
</body>
</html>

在服务器中输入 http://localhost:8080/index.html 即可看到效果。

上述代码，通过 fs.readFile()，读取文件系统中的文件，如果请求路径存在文件系统中，则读取内容并返回，若不存在，则返回 err。

# 实现文件服务器

结合 url 和 path 模块可以实现一个 文件服务器。

file_server.js

'use strict';

var fs = require('fs'),
    url = require('url'),
    path = require('path'),
    http = require('http');

// 从命令行参数获取root目录，默认是当前目录:
var root = path.resolve(process.argv[2] || '.');

console.log('Static root dir: ' + root);

// 创建服务器:
var server = http.createServer(function (request, response) {
    // 获得URL的path
    var pathname = url.parse(request.url).pathname;
    console.log('pathname: ', pathname)
    // 获得对应的本地文件路径
    var filepath = path.join(root, pathname);
    console.log('filepath: ', filepath)
    // 获取文件状态:
    fs.stat(filepath, function (err, stats) {
        if (!err && stats.isFile()) {
            // 没有出错并且文件存在:
            console.log('200 ' + request.url);
            // 发送200响应:
            response.writeHead(200);
            // 将文件流导向response:
            fs.createReadStream(filepath).pipe(response);
        } else {
            // 出错了或者文件不存在:
            console.log('404 ' + request.url);
            // 发送404响应:
            response.writeHead(404);
            response.end('404 Not Found');
        }
    });
});

server.listen(2000);

console.log('Server is running at http://127.0.0.1:2000/');

没有必要手动读取文件内容。由于response对象本身是一个Writable Stream，直接用pipe()方法就实现了自动读取文件内容并输出到HTTP响应。

在命令行直接运行 nodefile_server，将当前工作目录作为文件服务器，如果当前目录下有一个叫index.html的文件，在浏览器中输入http://localhost:2000/index.html即可访问。

在命令行运行node file_server.js /path/to/dir，把/path/to/dir改成你本地的一个有效的目录，然后在浏览器中输入http://localhost:2000/index.html。

在浏览器输入http://localhost:8080/时，会返回404，原因是程序识别出HTTP请求的不是文件，而是目录。

# 请求处理 GET

首先在 www 下新建一个 form.html

<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <meta http-equiv="X-UA-Compatible" content="ie=edge">
  <title>form</title>
</head>
<body>
  <form action="http://localhost:8080/form" method="get">
    username: <input type="text" name="username">
    password: <input type="password" name="password">
    <input type="submit" value="submit">
  </form>
</body>
</html>

在 server.js 中处理get请求:

const http = require('http')

const server = http.createServer((req, res) => {
  let url
  if (req.url.indexOf('?') > -1) {
    let GET = {}
    let args = req.url.split('?')
    url = args[0]
    let parms = args[1].split('&')
    for (let parm of parms) {
      const key = parm.split('=')[0]
      const value = parm.split('=')[1]
      GET[key] = value
    }
    console.log('query: ', GET)
  } else {
    url = req.url
  }
  console.log(url)
  res.end()
});

server.listen(8080)

此时使用 www/login.html 提交表单可以看到控制台输出。

# querystring

解析查询字符串有一个专门的模块，来帮我们解析地址栏的查询字符串(类似于 xxx=aa&xx=bb 格式的字符串)。

比如上述代码可以改写为:

const http = require('http')
const querystring = require('querystring')

const server = http.createServer((req, res) => {
  let url
  if (req.url.indexOf('?') > -1) {
    let GET = {}
    let args = req.url.split('?')
    url = args[0]
    GET = querystring.parse(args[1])
    console.log('query: ', GET)
  } else {
    url = req.url
  }
  console.log(url)
  res.end()
});

server.listen(8080)

# url

结合 url 模块，解析请求路径将会变得更加简单，上述代码可以改写为:

const http = require('http')
const url = require('url')

const server = http.createServer((req, res) => {

  console.log(pathName)
  console.log(GET)
  res.end()
}).listen(8080);

# 请求处理 POST

首先在 www 下新建一个 form.html

<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <meta http-equiv="X-UA-Compatible" content="ie=edge">
  <title>form</title>
</head>
<body>
  <form action="http://localhost:8080/form" method="post">
    title: <input type="text" name="username"><br>
    content: <textarea name="content" id="" cols="30" rows="10"></textarea>
    <input type="submit" value="submit">
  </form>
</body>
</html>

在 server.js 中处理post请求:

透过监听 data 接收数据，通过监听 end 完成数据的接收。

const http = require('http')
const querystring = require('querystring')

const server = http.createServer((req, res) => {
  let str = ''
  let i = 0
  // data -- 一段数据到达的时候触发(多次)
  req.on('data', (data) => {
    console.log(`第${++i}次收到数据`)
    str += data
  })
  // end -- 数据全部接收完成(一次)
  req.on('end', () => {
    let POST = querystring.parse(str)
    console.log(POST)
    res.end()
  })
}).listen(8080);

# 一个处理完整请求的服务器

结合之前学习的只是，可以搭建一个简单的处理完整请求的服务器:

const http = require('http')
const querystring = require('querystring')
const url = require('url')
const fs = require('fs')

const server = http.createServer((req, res) => {
  // GET
  let urlObj = url.parse(req.url, true)
  let pathName = urlObj.pathname
  const GET = urlObj.query

  // POST
  let str = ''
  req.on('data', (data) => {
    str += data
  })
  req.on('end', () => {
    let POST = querystring.parse(str)
    console.log(pathName, GET, POST)
  })

  // 文件请求
  var file_name = './www/' + req.url
  fs.readFile(file_name, (err, data) => {
    if (err) {
      res.write('404')
    } else {
      res.write(data)
    }
    res.end()
  })
}).listen(8080)

# 接口封装

通过之前的学习，我们可以封装一个简单的处理接口请求的服务器。

需求:

用户登录、注册

注册 `/user?act=reg&user=xxx&pwd=xxxx`

登录 `/user?act=login&user=xxx&pwd=xxx`

返回

{"ok": false, "msg": "原因"}
{"ok": true, "msg": "原因"}

对文件访问：

http://localhost:8080/1.html
http://localhost:8080/1.js
http://localhost:8080/1.jpg

对接口访问：

http://localhost:8080/user?act=xx...

接下来是完整的代码:

server.js

const http = require('http')
const fs = require('fs')
const querystring = require('querystring')
const urlLib = require('url')

var users={}   // {"blue": "123456", "zhangsan": "123456", "lisi": "321321"}

var server=http.createServer((req, res) => {
  // 解析数据
  var str=''
  req.on('data', data => {
    str += data
  })
  req.on('end', () => {
    var obj = urlLib.parse(req.url, true)

    const url = obj.pathname
    const GET = obj.query
    const POST = querystring.parse(str)

    if (url === '/user') {
      // 处理接口请求
      switch (GET.act) {
        case 'reg':
          if (users[GET.user]) {
            // 1.检查用户名是否已经有了
            res.write('{"ok": false, "msg": "此用户已存在"}')
          }else{
            // 2.注册成功，插入 users
            users[GET.user] = GET.pass
            res.write('{"ok": true, "msg": "注册成功"}')
          }
          break
        case 'login':
          if (users[GET.user] === null) {
            // 1.检查用户是否存在
            res.write('{"ok": false, "msg": "此用户不存在"}')
          } else if (users[GET.user] !== GET.pass) {
            // 2.检查用户密码
            res.write('{"ok": false, "msg": "用户名或密码有误"}')
          } else {
            // 3.登录成功
            res.write('{"ok": true, "msg": "登录成功"}')
          }
          break
        default:
          res.write('{"ok": false, "msg": "未知的act"}')
      }
      res.end()
    } else {
      // 读取文件
      var file_name = './www' + url
      fs.readFile(file_name, (err, data) => {
        if (err) {
          res.write('404')
        } else {
          res.write(data)
        }
        res.end()
      })
    }
  })
}).listen(8080)

src/ajax.js

function json2url(json){
  var arr=[];
  for(var name in json){
    arr.push(name+'='+json[name]);
  }
  return arr.join('&');
}

function ajax(json){
  json=json || {};
  if(!json.url)return;
  json.data=json.data || {};
  json.type=json.type || 'get';

  var timer=null;

  if(window.XMLHttpRequest){
    var oAjax=new XMLHttpRequest();
  }else{
    var oAjax=new ActiveXObject('Microsoft.XMLHTTP');
  }

  switch(json.type){
    case 'get':
      oAjax.open('GET',json.url+'?'+json2url(json.data),true);
      oAjax.send();
      break;
    case 'post':
      oAjax.open('POST',json.url,true);
      oAjax.setRequestHeader('Content-Type','application/x-www-form-urlencoded');
      oAjax.send(json2url(json.data));
      break;
  }

  oAjax.onreadystatechange=function(){
    if(oAjax.readyState==4){
      clearTimeout(timer);
      if(oAjax.status>=200 && oAjax.status<300 || oAjax.status==304){
        json.success && json.success(oAjax.responseText);
      }else{
        json.error && json.error(oAjax.status);
      }
    }
  };
}
