    # 小程序插件 - 同声传译

    • 小程序同声传译插件 GitHub
    • 小程序同声传译插件官方介绍

    # 配置插件

    在小程序后台 "设置->第三方服务->插件管理" 添加此插件：

    如果使用原生小程序开发，在 app.json 中配置：

    {
  "plugins": {
    "WechatSI": {
      "version": "0.0.6",
      "provider": "wx069ba97219f66d99"
    }
  }
}

    如果使用 uni-app 开发，在 manifest.json 中配置：

    {
  "mp-weixin" : {
    "plugins": {
      "WechatSI": {
        "version": "0.0.6",
        "provider": "wx069ba97219f66d99"
      }
    }
  }
}

    # 使用插件

    首先引入插件：

    const plugin = requirePlugin("WechatSI")

    # 语音输入

    获取全局语音输入对象：

    const manager = plugin.getRecordRecognitionManager()

    manager 用如下实例方法

    recordRecoManager 对象的实例方法：

    • start(option) 开始识别
    • stop 结束识别

    start 的option参数包括

    • duration Number 可选，默认60000。指定录音的时长，单位ms，最大为60000。如果传入了合法的 duration ，在到达指定的 duration 后会自动停止录音
    • lang String 可选，默认zh_CN。识别的语言，目前支持zh_CN en_US zh_HK

    manager 有如下事件：

    • onStart 正常开始录音识别时会调用此事件
    • onRecognize 有新的识别内容返回，则会调用此事件
    • onStop 识别结束事件
    • onError 识别错误事件

    使用 uni-app 进行开发，测试 Demo 如下

    <template lang='pug'>
  .main
    button(@tap='record') 开始录音
    button(@tap='stop') 结束录音
    text {{msg}}{{currMsg}}
</template>

<script>
const plugin = requirePlugin("WechatSI")
const manager = plugin.getRecordRecognitionManager()
export default {
  data () {
    return {
      msg: '',
      currMsg: ''
    }
  },
  onLoad () {
    manager.onRecognize = res => {
      console.log(res);
      this.currMsg = res.result
    }
    manager.onStop = res => {
      console.log(res);
      this.msg = this.msg + res.result
      this.currMsg = ''
    }
    manager.onError = e => {
      console.log(e);
    }
  },
  methods: {
    record () {
      manager.start()
    },
    stop () {
      manager.stop()
    }
  }
}
</script>

    # 讯飞集成

    • 讯飞开放平台： https://www.xfyun.cn/
    • 讯飞语音识别接口文档： https://doc.xfyun.cn/rest_api/语音听写.html

    # 接口调用

    接口地址：

    POST http://api.xfyun.cn/v1/service/v1/iat HTTP/1.1
Content-Type:application/x-www-form-urlencoded; charset=utf-8

    请求头中(全部必填)：

    • X-Appid string 讯飞开放平台注册申请应用的应用ID(appid)
    • X-CurTime string 当前UTC时间戳，从1970年1月1日0点0 分0 秒开始到现在的秒数
    • X-Param string 相关参数JSON串经Base64编码后的字符串
    • X-CheckSum string 令牌，计算方法：MD5(apiKey + curTime + param)，三个值拼接的字符串，进行MD5哈希计算（32位小写），其中apiKey由讯飞提供，调用方管理。
    • Content-Type 填写 application/x-www-form-urlencoded

    其中 X-Param 为以下字段的计算结果

    • engine_type string 必填。默认sms8k。 引擎类型，可选值：sms16k（16k采样率普通话音频）、sms8k（8k采样率普通话音频）等。
    • aue string 必填。默认raw。 音频编码，可选值：raw（未压缩的pcm或wav格式）、speex（speex格式）、speex-wb（宽频speex格式）。
    • speex_size string 选填。默认60。 speex音频帧率，speex音频必传。
    • scene string 选填。默认main。 情景模式。 如需使用热词功能，必须指定scene=main。
    • vad_eos string 选填。默认2000。 后端点检测（单位：ms），默认1800，取值范围0-10000。

    X-Param 生成示例：

    // 原始JSON串：
{
  "engine_type": "sms16k",
  "aue": "raw"
}
// BASE64编码（即X-Param）：
eyJlbmdpbmVfdHlwZSI6ICJzbXMxNmsiLCJhdWUiOiAicmF3In0=

    请求体中：

    • audio: 需要识别的声音，需要转化为 base64 格式

    # 调用示例

    在使用 uni-app 进行小程序开发中的示例：

    <template>
<view>
  <button @tap="startRecord">开始录音</button>
  <button @tap="endRecord">停止录音</button>
  <button @tap="playVoice">播放录音</button>
  <button @click="fun">语音转文字</button>
</view>
</template>

<script>
import { Base64 } from 'js-base64'
var md5 = require('md5-js');

const recorderManager = uni.getRecorderManager();
const innerAudioContext = uni.createInnerAudioContext();

innerAudioContext.autoplay = true;

export default {
  data: {
    text: 'uni-app',
    voicePath: ''
  },
  onLoad() {
    let self = this;
    recorderManager.onStop(function (res) {
      console.log('recorder stop' + JSON.stringify(res));
      self.voicePath = res.tempFilePath;
    });
  },
  methods: {
    startRecord() {
      console.log('开始录音');
      recorderManager.start();
    },
    endRecord() {
      console.log('录音结束');
      recorderManager.stop();
    },
    playVoice() {
      console.log('播放录音');
      if (this.voicePath) {
        console.log(this.voicePath);
        innerAudioContext.src = this.voicePath;
        innerAudioContext.play();
      }
    },
    fun () {
      let url = this.voicePath
      let vm = this
      let appid = '5bdc9c5a'
      let originParams = {
        "engine_type": "sms16k",
        "aue": "raw"
      }
      let apikey = 'b805fba9b20ffa2430cf096162a7a551'
      console.log(JSON.stringify(originParams));
      let xParam = Base64.encode(JSON.stringify(originParams))
      console.log(xParam);
      let curTime = parseInt(new Date().getTime() / 1000)
      console.log(curTime);
      let checkSum = md5(apikey+curTime+xParam)
      console.log(checkSum);
      uni.request({
        url: url,
        method:'GET',
        responseType: 'arraybuffer',
        success: ress => {
          let base64 = wx.arrayBufferToBase64(ress.data); //把arraybuffer转成base64
          vm.$axios.base({
            url: 'http://api.xfyun.cn/v1/service/v1/iat',
            data: {
              audio: base64
            },
            header: {
              'Content-Type': 'application/x-www-form-urlencoded',
              'X-Appid': appid,
              'X-CurTime': curTime,
              'X-Param': xParam,
              'X-CheckSum': checkSum
            }
          }, 'POST').then(res => {
            console.log(res);
          }).catch(e => {
            console.log(e);
          })
        }
      })
    }
  }
}
</script>
