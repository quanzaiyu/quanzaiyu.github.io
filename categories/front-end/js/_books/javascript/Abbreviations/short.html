# 短路求值简写

# 短路或

# 短路或判断非空值

将变量值分配给另一个变量时，你可能需要确保源变量不为 null、undefined 或者不为空。此时可以使用多个条件写一个很长的 if 语句，也可以使用短路求值的简写办法。

if (variable1 !== null || variable1 !== undefined || variable1 !== '') {
     let variable2 = variable1;
}

简写为

const variable2 = variable1  || 'new';

# 短路或求值

let a = 1, b = 2

if (a) {
  d = a
} else {
  d = b
} // 1

简写为

let a = 1, b = 2

let d = a || b // 1

# 短路与

# 短路与求值

let a = 1, b = 2

if (a) {
  c = b
} else {
  c = a
} // c is 2

简写为

let a = 1, b = 2

let c = a && b // c is 2
