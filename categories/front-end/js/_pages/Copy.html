              # JS 重难点解析 - 对象及数组的深拷贝和浅拷贝

              以下实例, 同时适用于对象和数组的操作。

              # 一、对象引用

              赋值运算符不会创建一个对象的副本，它只分配一个引用。

              Javascript 的对象只是指向内存中某个位置的指针。这些指针是可变的，也就是说，它们可以重新被赋值。因此，单单复制这个指针的结果是，有两个指针指向内存中的同一块地址。

              let obj = {
  a: 1,
  b: 2,
};

let copy = obj;

obj.a = 5
console.log(copy.a)  // 5
console.log(obj === copy) // true

              # 二、通过 for..in 循环遍历复制对象

              复制对象的原始方法是循环遍历原始对象，然后一个接一个地复制每个属性。

              function copy(mainObj) {
  let objCopy = {}; // objCopy 将存储 mainObj 的副本
  let key;

  for (key in mainObj) {
    objCopy[key] = mainObj[key]; // 将每个属性复制到objCopy对象
  }
  return objCopy;
}

let mainObj = {
  a: 2,
  b: 5,
  c: {
    x: 7,
    y: 4,
  }
}

let copyObj = copy(mainObj) // { a: 2, b: 5, c: { x: 7, y: 4, } }

mainObj.a = 3
console.log(mainObj) // { a: 3, b: 5, c: { x: 7, y: 4, } }
console.log(copyObj) // { a: 2, b: 5, c: { x: 7, y: 4, } }

mainObj.c.x = 10
console.log(mainObj) // { a: 3, b: 5, c: { x: 10, y: 4, } }
console.log(copyObj) // { a: 2, b: 5, c: { x: 10, y: 4, } }

              存在的问题：上面的代码，如果原始对象中的一个属性本身就是一个对象，那么副本和原始对象之间将共享这个对象，从而使其各自的属性指向同一个对象。

              # 三、使用 Object.assign() 浅拷贝对象

              当拷贝源对象的顶级属性被复制而没有任何引用，并且拷贝源对象存在一个值为对象的属性，被复制为一个引用时，那么我说这个对象被浅拷贝。如果拷贝源对象的属性值是对象的引用，则只将该引用值复制到目标对象。

              浅层复制将复制顶级属性，但是嵌套对象将在原始（源）对象和副本（目标）对象之间是共享。

              使用 Object.assign() 和一个空对象({})来创建原始对象的浅拷贝。

              const shallowClone = obj => Object.assign({}, obj);

let obj = {
  a: {
    b: 0,
    c: {
      d: 1
    }
  },
  e: 5
}

let cloneObj = shallowClone(obj)
console.log(cloneObj === obj) // false
console.log(cloneObj.a === obj.a) // true

              # 四、使用 JSON.stringify() 深拷贝对象

              深度拷贝将拷贝遇到的每个对象。副本和原始对象不会共享任何东西，所以它将是原件的副本。

              使用 JSON.stringify() 将对象字符串化，再通过 JSON.parse() 创建一个新的对象，这种方式是深拷贝对象。

              const deepClone = (obj) => JSON.parse(JSON.stringify(obj))

let obj = {
  a: {
    b: 0,
    c: {
      d: 1
    }
  },
  e: 5
}

let cloneObj = deepClone(obj)
console.log(cloneObj === obj) // false
console.log(cloneObj.a === obj.a) // false

              不幸的是，此方法不能用于复制用户定义的对象方法。

              let obj = {
  name: 'scotch.io',
  exec: function exec() {
    return true;
  }
}

let method1 = Object.assign({}, obj);
let method2 = JSON.parse(JSON.stringify(obj));

console.log(method1); //Object.assign({}, obj)
/* result
{
  exec: function exec() {
    return true;
  },
  name: "scotch.io"
}
*/

console.log(method2); // JSON.parse(JSON.stringify(obj))
/* result
{
  name: "scotch.io"
}
*/

              # 五、复制循环引用对象

              循环引用对象是具有引用自身属性的对象。

              # JSON.parse(JSON.stringify(obj))

              // 循环引用对象
let obj = {
  a: 'a',
  b: {
    c: 'c',
    d: 'd',
  },
}

obj.c = obj.b;
obj.e = obj.a;
obj.b.c = obj.c;
obj.b.d = obj.b;
obj.b.e = obj.b.c;

let newObj = JSON.parse(JSON.stringify(obj));

console.log(newObj);

              结果报错:

              Uncaught TypeError: Converting circular structure to JSON
    at JSON.stringify (<anonymous>)
    at index.html:31

              总结: JSON.parse(JSON.stringify(object)) 不能复制循环引用对象。

              # Object.assign()

              // 循环引用对象
let obj = {
  a: 'a',
  b: {
    c: 'c',
    d: 'd',
  },
}

obj.c = obj.b;
obj.e = obj.a;
obj.b.c = obj.c;
obj.b.d = obj.b;
obj.b.e = obj.b.c;

let newObj2 = Object.assign({}, obj);

console.log(newObj2);

              运行结果:

              Object.assign() 适用于浅拷贝循环引用对象，但不适用于深度拷贝。但能够随意浏览浏览器控制台上的循环引用对象树。

              # 六、使用扩展操作符(...)复制

              ES6 提供一种更加简便的复制方式: 扩展操作符(...)

              // 对象
var obj = { foo: "foo", bar: "bar" }
var copy = { ...obj } // { foo: "foo", bar: "bar" }

// 数组
const array = [
  "a",
  "c",
  "d", {
    four: 4
  }
];
const newArray = [...array]
console.log(newArray === array) // false
console.log(newArray[3] === array[3]) // true

              但是同样地, 如果元素本身就是引用类型, 仍然指向的是同一片内存。
