# JS 重难点解析 - 作用域与闭包

# 闭包的高级应用

# Function构造器生成的函数，在全局作用域中被创建

普通闭包 :

var n = 1;
function f(){
    var n = 2;
    function e(){
        return ++n;
    }
    return e;
}
let fn = f()
fn() // 3
fn() // 4
fn() // 5

let fn2 = f()
fn2() // 3
fn2() // 4
fn2() // 5

使用 Function 构造的闭包:

var n = 1;
function f(){
    var n = 2;
    var e = new Function("return ++n;");
    return e;
}

let fn = f()
fn() // 2
fn() // 3
fn() // 4

let fn2 = f()
fn2() // 2
fn2() // 3
fn2() // 4

注意到，使用 Function 构造的闭包，使用的是全局作用域中的 n ，而不是外层函数中的 n，因此，此闭包是在全局作用域中创建的。
