# 集成 Redux

React Nagivation 可以很好地集合 Redux 使用，以方便做一些全局的状态管理。

安装相关的库：

$ yarn add redux react-redux

本文使用的版本：

"react-redux": "^6.0.1",
"redux": "^4.0.1",

# 修改根组件

App.js

import React, {Component} from 'react';
import {createAppContainer} from "react-navigation"
import AppNavigations from './src/navigators/AppNavigations'
import {Provider} from "react-redux"
import store from './src/stores'

let Navigation = createAppContainer(AppNavigations);

type Props = {};
export default class App extends Component<Props> {
  render() {
    return (
      <Provider store={store}>
        <Navigation />
      </Provider>
    );
  }
}

这里使用了 Provider，用于自上而下向所有组件提供全局store

# 在组件中使用

import React from 'react';
import {
  Button,
  Text,
  View,
} from "react-native";
import { connect } from 'react-redux'

class UserInfo extends React.Component {
  constructor(props) {
    super(props);
  }

  render() {
    const { app, dispatch } = this.props // 将数据取出
    return (
      <View>
        <Button onPress={() => {
          dispatch({
            type: 'SET_APP_NAME',
            text: 'react native'
          })
        }} title={'SET_APP_NAME'}/>
        <Text>{ app }</Text>
      </View>
    );
  }
}

// 将数据连接到组件
export default connect(state => ({
  app: state.app
}))(UserInfo);
