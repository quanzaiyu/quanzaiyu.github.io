# 轮播图

  • react-native-looped-carousel

安装：

$ yarn add react-native-looped-carousel

使用：

import React,{Component} from 'react';
import { rem, deviceWidthDp } from '../../utils/px2dp'
import {
  StyleSheet,
  Text,
  View,
  Image
} from 'react-native';
import Carousel from "react-native-looped-carousel"

class Home extends Component{
  render(){
    return (
      <View>
        {Home.renderCarousels()}
      </View>
    );
  }

  // 轮播图
  static renderCarousels() {
    return (
      <View>
        <Carousel
          delay={2000}
          style={styles.wrapper}
          autoplay
          pageInfo
          onAnimateNextPage={(p) => console.log(p)}
        >
          <View style={[{ backgroundColor: '#BADA55' }, styles.wrapper]}><Text>1</Text></View>
          <View style={[{ backgroundColor: 'red' }, styles.wrapper]}><Text>2</Text></View>
          <View style={[{ backgroundColor: 'blue' }, styles.wrapper]}><Text>3</Text></View>
        </Carousel>
      </View>
    )
  }
}

//设置样式
const styles = StyleSheet.create({
  wrapper: {
    width: rem(10),
    height: rem(10)
  },
});

export default Home
