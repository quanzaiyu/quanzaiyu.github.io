# Redux 触发视图更新

如果在Redux中更新了数据，而在React中却未触发render，可能存在以下问题：

注意 reducer 的书写，注意不要对传来的state直接修改，state是对象引用类型(指向堆）

错误的写法示例：

modules/user.js

const app = (state = {}, action) => {
  if (action.module === 'app') {
    switch (action.type) {
      case 'name':
        state.name = action.data
        break
      default:
        break
    }
  }
  return state
}

export default app

正确的写法为：

modules/user.js

const app = (state = {}, action) => {
  let stateCopy = {...state}
  if (action.module === 'app') {
    switch (action.type) {
      case 'name':
        stateCopy.name = action.data
        break
      default:
        break
    }
  }
  return stateCopy
}

export default app

reducer.js

import { combineReducers } from 'redux'

import app from './modules/app'
import user from './modules/user'

const App = combineReducers({
  user
})

export default App

action.js

export const setData = (module, type, data)=> {
  return {
    module, type, data
  }
}

组件中：

"use strict";

import React from 'react';
import {
  Button,
  Text,
  View,
} from "react-native"
import { connect } from 'react-redux'
import { setData } from "./actions"

class UserInfo extends React.PureComponent {
  constructor(props) {
    super(props);
  }

  render() {
    const { store } = this.props
    return (
      <View>
        <Button onPress={() => { this.setName() }} title={'SET_APP_NAME'}/>
        <Text>{store.app.name}</Text>
      </View>
    );
  }

  setName() {
    const { store, dispatch } = this.props
    dispatch(setData('app', 'name', 'hello'))
  }
}

export default connect(state => ({ store: state }))(UserInfo);
