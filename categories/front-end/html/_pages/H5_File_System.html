                            # 深入理解 HTML5 - 文件系统

                            web应用很酷，但是相对于桌面应用来说，它们有比较显著的弱点：它们无法在一个有层次的文件夹结构体即文件系统中互动和组织。 幸运的是，如果我们使用Filesystem API，我们可以做到。这个API帮助我们控制私有的本地文件系统“沙箱(sandbox)"，在这里我们可以读和写文件，创建和排列文件夹。

                            # 申请空间

                            为了进行数据的存储，必须要向浏览器进行申请，如果是永久存储还会向用户进行询问，只有同意后才会继续执行。

                            1. 检查浏览器是否支持FileSystem
                            2. 申请配额
                            3. 成功与否的处理
                            console.info(window.TEMPORARY);  //0  临时
console.info(window.PERSISTENT); //1  持久

window.requestFileSystem = window.requestFileSystem || window.webkitRequestFileSystem;

window.onload = function() {
  // 检查浏览器是否支持FileSystem
  if (typeof window.requestFileSystem != "undefined") {
    window.requestFileSystem(
      window.PERSISTENT, // PERSISTENT(永久) or TEMPORARY(临时)
      1024*1024, // 申请空间的大小, 1M
      onInitFs, // 成功后的回调函数
      errorHandler // 错误后的回调函数
    );
  } else {
    console.log("Sorry, your browser does not support FileSystem.");
  }
}

                            应用在进行文件操作时，必须先请求系统上存储数据配额，尤其针对PERSISTENT持久存储需要用户点击确认，如果用户拒绝文件访问失败

                            详解 requestFileSystem

                            window.requestFileSystem(type, size, successCallback, opt_errorCallback)

                            # 成功与否的回调

                            在成功回调中获得FileSystem对象fs

                            function onInitFs(fs) {
  // ...
}
function errorHandler(err) {
  console.info('创建文件失败');
  console.info(err);
}
                            • 成功调用 onInitFs, 包含一个参数fs, 其为 DOMFileSystem 对象
                            • 失败则调用 errorHandler

                            但是这是有个问题，这样做的话每次加载页面都会申请，这肯定不是我们想要的，我们要的是在有数据的时候就可以读取数据。

                            # 创建文件夹

                            以下代码均为成功回调的逻辑处理

                            fs.root.getDirectory('Documents', {create: true}, function(dirEntry) {
  console.log('You have just created the ' + dirEntry.name + ' directory.');
}, errorHandler);

                            getDiretory() 方法用来读和创建目录。

                            • 第一个参数，你可以传递一个名字或者路径来寻找或者创建
                            • 第二个参数为true，因为我们需要创建一个目录 - 不是读一个已存在的目录
                            • 第三个参数为发生错误时的回调

                            我们创建了一个目录，接着我们可以继续创建一个子目录 Documents/Music。但是如果你想深度创建一个子目录，使用多级目录那么怎么做呢？ 如果你使用Documents/Music/Nature/Sky作为路径参数，你会得到错误，因为你不能创建一个没有父目录的目录。解决方式是一个一个的创建。但是这样很低效并且麻烦。更好的解决方式：创建一个方法用来自动创建目录：

                            function createDir(rootDir, folders) {
  rootDir.getDirectory(folders[0], {create: true}, function(dirEntry) {
    if (folders.length) {
      createDir(dirEntry, folders.slice(1));
    }
  }, errorHandler);
};

createDir(fs.root, 'Documents/Images/Nature/Sky/'.split('/'));

                            fs.root 获取根文件系统, 其是 DirectoryEntry 对象, 每次创建文件夹后将创建的文件夹作为新的入口文件夹

                            # 获取文件入口

                            FileSystem使用了特殊的文件系统和沙盒模式，在电脑上或者其他web中是无法访问沙盒中的文件的 ，只能用对应的格式去访问。

                            在浏览器中输入：

                            • filesystem:http://localhost/persistent/ 获取永久空间
                            • filesystem:http://localhost/temporary/ 获取临时空间

                            比如: filesystem:http://localhost/persistent/test.txt, 获取 /test.txt 文件, 参考下文中创建文件

                            补充: LocalURL 本地路径URL

                            说明：可在html页面中直接访问本地资源，以 file:/// 开头，后面跟随系统的绝对路径。示例：file:///D:/res/hello.html。沙盒系统以filesystem:http://开头

                            补充: RemoteURL 网络路径URL

                            说明：可在html页面中以网络资源模式访问本地资源，以 http:// 开头，后面跟随相对路径。示例：http://localhost:13131/_www/res/icon.png。

                            # 读取文件系统

                            接下来我们需要检查我们的文件系统。我们创建一个DirectoryReader对象，使用ReadEntries()方法来读取目录中的内容。

                            fs.root.getDirectory('/', {}, function(dirEntry){
  let dirReader = dirEntry.createReader();
  dirReader.readEntries(function(entries) {
    for(let i = 0; i < entries.length; i++) {
      let entry = entries[i];
      if (entry.isDirectory){
        console.log('Directory: ' + entry.fullPath);
      }
      else if (entry.isFile){
        console.log('File: ' + entry.fullPath);
      }
    }
  }, errorHandler);
}, errorHandler);

                            # 删除文件夹

                            有两种方式来删除一个DirectoryEntry: remove() 和 removeRecursively()。

                            区别在于: 第一个删除需要被删除文件夹为空，否则会得到错误。

                            // 安全删除
fs.root.getDirectory('Documents/Music', {}, function(dirEntry) {
  dirEntry.remove(function(){
    console.log('Directory successfully removed.');
  }, errorHandler);
}, errorHandler);

// 递归删除
fs.root.getDirectory('Documents/Music', {}, function(dirEntry) {
  dirEntry.removeRecursively(function(){
    console.log('Directory successufully removed.');
  }, errorHandler);
}, errorHandler);

                            # 创建文件

                            fs.root.getFile('test.txt', {create: true, exclusive: true}, function(fileEntry) {
  console.log('A file ' + fileEntry.name + ' was created successfully.');
}, errorHandler);

                            getFile方法的第一个参数可以是绝对或者相对路径，但是必须是合法的。例如，没有父目录创建一个文件会得到一个错误。第二个参数是一个对象说明，如果文件不存在的话描述功能行为。在这个例子中：create:true 表示如果文件不存则创建一个文件，如果存在则抛出错误 exclusive:true。如果 create:false，简单取得文件并返回。

                            fs.root.getFile('test.txt', {create: false}, function(fileEntry) {
  fileEntry.createWriter(function(fileWriter) {
    let blob = new Blob(["Hello world"], {type : 'text/plain'})
    fileWriter.write(blob);
  }, errorHandler);
}, errorHandler);

                            以上代码中，我们返回了 test.txt 文件，创建了FileWriter对象。我们然后通过创建一个新的BlobBuilder对象添加内容并且使用了FileWriter的write()方法。

                            # 读取文件内容

                            调用getFile()方法只会返回FileEntry对象。并不返回文件的内容。因此，如果我们想读出文件内容，我们需要使用File对象和FileReader对象。

                            fs.root.getFile('test.txt', {}, function(fileEntry) {
  fileEntry.file(function(file) {
    let reader = new FileReader();
    reader.onloadend = function(e) {
      console.log(this.result);
    };
    reader.readAsText(file);
  }, errorHandler);
}, errorHandler);

                            # 追加文件内容

                            我们已经写入了文件一些内容，但是如果以后添加更多内容呢？为了添加内容到已存在的文件，又需要调用FileWriter。我们可以使用seek()方法重新将writer添加到文件。seek接受字节偏移(byte offset)这个参数，并且设置file writer的位置。

                            fs.root.getFile('test.txt', {create: false}, function(fileEntry) {
  fileEntry.createWriter(function(fileWriter) {
    fileWriter.seek(fileWriter.length);
    let blob = new Blob(["Append text"], {type : 'text/plain'})

                            # 删除文件

                            如果需要删除文件，我们调用entry.remove()。一个参数是当文件被成功删除后调用。第二个参数是一个可选的错误回调函数。

                            fs.root.getFile('test.txt', {create: false}, function(fileEntry) {
  fileEntry.remove(function() {
    console.log('File successufully removed.');
  }, errorHandler);
}, errorHandler);

                            # 拷贝、移动和重命名

                            FileEntry和DirectoryEntry分享同一个API方法来拷贝，移动和重命名。这里有俩个方法你可以使用进行操作：copyTo()和moveTo()。他们都接受同样的参数：

                            copyTo(parentDirEntry, opt_newName, opt_successCallback, opt_errorCallback); // 拷贝
moveTo(parentDirEntry, opt_newName, opt_successCallback, opt_errorCallback); // 移动与重命名

                            拷贝

                            function copy(currDir, srcEntry, destDir) {
  currDir.getFile(srcEntry, {}, function(fileEntry) {
    currDir.getDirectory(destDir, {}, function(dirEntry) {
      fileEntry.copyTo(dirEntry);
    }, errorHandler);
  }, errorHandler);
}

copy(fs.root, 'test.txt', 'Documents/');

                            移动

                            function move(currDir, srcEntry, dirName) {
  currDir.getFile(srcEntry, {}, function(fileEntry) {
    currDir.getDirectory(dirName, {}, function(dirEntry) {
      fileEntry.moveTo(dirEntry);
    }, errorHandler);
  }, errorHandler);
}

move(fs.root, 'test.txt', 'Documents/');

                            重命名

                            function rename(currDir, srcEntry, newName) {
  currDir.getFile(srcEntry, {}, function(fileEntry) {
    fileEntry.moveTo(currDir, newName);
  }, errorHandler);
}

rename(fs.root, 'test.txt', 'test1.txt');

                            # 获取所有文件及文件夹

                            通过递归的方式获取所有文件及文件夹

                            function findAll (root, fs) {
  root.getDirectory(fs, {}, function(dirEntry){
    let dirReader = dirEntry.createReader();
    dirReader.readEntries(function(entries) {
      for(let i = 0; i < entries.length; i++) {
        let entry = entries[i];
        if (entry.isDirectory){
          console.log('Directory: ' + entry.fullPath);
          findAll(entry)
        }
        else if (entry.isFile){
          console.log('File: ' + entry.fullPath);
        }
      }
    }, errorHandler);
  }, errorHandler);
}

findAll(fs.root, '/')
