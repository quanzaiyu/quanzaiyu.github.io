# RN第三方库 - 渐变

# 线性渐变

使用 react-native-linear-gradient 创建 React Native 线性渐变.

# 安装

$ yarn add react-native-linear-gradient
$ react-native link react-native-linear-gradient

也可手动配置:

在 android/settings.gradle 中:

include ':react-native-linear-gradient'
project(':react-native-linear-gradient').projectDir = new File(rootProject.projectDir, '../node_modules/react-native-linear-gradient/android')

在 android/app/build.gradle 中

dependencies {
  implementation project(':react-native-linear-gradient')
}

在 MainActivity.java (react-native < 0.29) 或 MainApplication.java (react-native >= 0.29) 中:

//...
import com.BV.LinearGradient.LinearGradientPackage;
//...
@Override
protected List<ReactPackage> getPackages() {
  return Arrays.<ReactPackage>asList(
    new MainReactPackage(),
    new LinearGradientPackage()
  );
}

# 使用

普通按钮示例:

import React, { Component } from 'react'
import { View, Text, StyleSheet } from 'react-native'
import LinearGradient from "react-native-linear-gradient";

export default class App extends Component {
  constructor (props) {
    super(props)
  }
  render () {
    return (
      <View style={{flex: 1, alignItems: 'center'}}>
        <View style={styles.container}>
          <LinearGradient colors={['#4c669f', '#3b5998', '#192f6a']} style={styles.linearGradient}>
            <Text style={styles.buttonText}>
              Sign in with Facebook
            </Text>
          </LinearGradient>
        </View>
        <View style={styles.container}>
          <LinearGradient start={{x: 0, y: 0}} end={{x: 1, y: 0}} colors={['#4c669f', '#3b5998', '#192f6a']} style={styles.linearGradient}>
            <Text style={styles.buttonText}>
              Sign in with Facebook
            </Text>
          </LinearGradient>
        </View>
        <View style={styles.container}>
          <LinearGradient
              start={{x: 0.0, y: 0.25}} end={{x: 0.5, y: 1.0}}
              locations={[0,0.5,0.6]}
              colors={['#4c669f', '#3b5998', '#192f6a']}
              style={styles.linearGradient}>
            <Text style={styles.buttonText}>
              Sign in with Facebook
            </Text>
          </LinearGradient>
        </View>
      </View>
    )
  }
}

// Later on in your styles..
const styles = StyleSheet.create({
  container: {
    height: 50,
    width: '80%',
    marginTop: 5
  },
  linearGradient: {
    flex: 1,
    paddingLeft: 15,
    paddingRight: 15,
    borderRadius: 5
  },
  buttonText: {
    fontSize: 18,
    fontFamily: 'Gill Sans',
    textAlign: 'center',
    margin: 10,
    color: '#ffffff',
    backgroundColor: 'transparent',
  },
});

渐变动画示例:

import React, { Component } from 'react'
import { View, Text, StyleSheet } from 'react-native'
import LinearGradient from "react-native-linear-gradient";

function incrementColor(color, step) {
  const intColor = parseInt(color.substr(1), 16);
  const newIntColor = (intColor + step).toString(16);
  return `#${'0'.repeat(6 - newIntColor.length)}${newIntColor}`;
};

export default class App extends Component {
  constructor (props) {
    super(props)
    this.state = {
      count: 0,
      colorTop: '#000000',
      colorBottom: '#cccccc'
    }
  }
  componentDidMount() {
    setInterval(() => {
      this.setState({
        count: this.state.count + 1,
        colorTop: incrementColor(this.state.colorTop, 1),
        colorBottom: incrementColor(this.state.colorBottom, -1),
      });
    }, 20);
  }

  render () {
    return (
      <View style={{flex: 1, alignItems: 'center'}}>
        <View style={styles.container}>
          <LinearGradient
              colors={[this.state.colorTop, this.state.colorBottom]}
              style={styles.gradient} />
          <Text style={{color: this.state.colorTop}}>{this.state.colorTop}</Text>
          <Text style={{color: this.state.colorBottom}}>{this.state.colorBottom}</Text>
        </View>
      </View>
    )
  }
}

// Later on in your styles..
const styles = StyleSheet.create({
  container: {
    height: 50,
    width: '80%',
    marginTop: 5
  },
  gradient: {
    width: 200,
    height: 200,
  },
});

# props

  • colors 渐变颜色数组, 比如 ['#4c669f', '#3b5998', '#192f6a']
  • start 渐变起点 { x: number, y: number }
  • end 渐变终点 { x: number, y: number }
  • locations 设置每个颜色的终点数组, 数值为 0~1, 比如 [0,0.5,0.6]

# 径向渐变

使用 react-native-radial-gradient 创建 React Native 径向渐变。

# 安装

$ yarn add react-native-radial-gradient
$ react-native link react-native-radial-gradient

也可手动配置:

在 android/settings.gradle 中:

include ':react-native-radial-gradient'
project(':react-native-radial-gradient').projectDir = new File(rootProject.projectDir, '../node_modules/react-native-radial-gradient/android')

在 android/app/build.gradle 中

dependencies {
  implementation project(':react-native-radial-gradient')
}

在 MainActivity.java (react-native < 0.29) 或 MainApplication.java (react-native >= 0.29) 中:

//...
import com.surajit.rnrg.RNRadialGradientPackage;
//...
@Override
protected List<ReactPackage> getPackages() {
  return Arrays.<ReactPackage>asList(
    new MainReactPackage(),
    new RNRadialGradientPackage(),
  );
}

注意构建工具的版本支持:

android/build.gradle

ext {
    compileSdkVersion   = 26
    targetSdkVersion    = 26
    buildToolsVersion   = "26.0.2"
}

# 使用

import React, { Component } from 'react'
import { View, Text, StyleSheet } from 'react-native'
import RadialGradient from "react-native-radial-gradient";

export default class App extends Component {
  render () {
    return (
        <View style={{flex: 1, alignItems: 'center'}}>
          <View style={{ height: 50, width: '80%', marginTop: 5 }}>
            <RadialGradient style={{ width: 200, height: 200, }}
                            colors={['black','green','blue','red']}
                            stops={[0.1,0.4,0.3,0.75]}
                            center={[100,100]}
                            radius={200}>
              <Text style={{color: '#f00'}}>Hello</Text>
            </RadialGradient>
          </View>
        </View>
    )
  }
}

# props

  • colors 渐变颜色数组, 比如 ['#4c669f', '#3b5998', '#192f6a']
  • stops 设置每个颜色的终点数组, 数值为 0~1, 比如 [0,0.5,0.6]
  • center 渐变中心坐标, 比如 [100,100]
  • radius 辐射半径, 默认 min(width/2,height/2)

# 文字渐变

使用 react-native-text-gradient

# 安装

$ yarn add react-native-text-gradient
$ react-native link react-native-text-gradient

也可手动配置:

在 android/settings.gradle 中:

include ':react-native-text-gradient'
project(':react-native-text-gradient').projectDir = new File(rootProject.projectDir, '../node_modules/react-native-text-gradient/android')

在 android/app/build.gradle 中

dependencies {
  implementation project(':react-native-text-gradient')
  compile project(':react-native-text-gradient')
}

在 MainActivity.java (react-native < 0.29) 或 MainApplication.java (react-native >= 0.29) 中:

//...
import iyegoroff.RNTextGradient.RNTextGradientPackage;
//...
@Override
protected List<ReactPackage> getPackages() {
  return Arrays.<ReactPackage>asList(
    new MainReactPackage(),
    new RNTextGradientPackage(),
  );
}

# 使用

用法类似于 Text & LinearGradient 的合集

import React, { Component } from 'react'
import { View, StyleSheet } from 'react-native'
import { LinearTextGradient } from 'react-native-text-gradient';

export default class App extends Component {
  render () {
    return (
        <View style={{flex: 1, alignItems: 'center'}}>
            <LinearTextGradient
                style={{ fontWeight: 'bold', fontSize: 72 }}
                locations={[0, 1]}
                colors={['red', 'blue']}
                start={{ x: 0, y: 0 }}
                end={{ x: 1, y: 0 }}
            >
              THIS IS TEXT GRADIENT
            </LinearTextGradient>
        </View>
    )
  }
}

在 rn >= 0.56.0 中, 第三方 类Text组件 会报错 Invariant Violation: Text strings must be rendered within a <Text> component., 等待官方修复 pull request: Allow third party react-native components to render strings (2019.2.26记)
