# 分支结构

# if-else

pug

-var a = 5
if a == 1
  p 1
else if a == 2
  p 2
else
  p false

html

<p>false</p>

# unless

unless语句当表达式为false的时候执行

pug

-var istrue=false
unless istrue
  p hello

html

<p>hello</p>

# case-when-default

pug

-var name='pug'
case name
  when 'java': p Hi,java
  when 'pug': p Hi,pug
  default: p Hi

html

<p>Hi,pug</p>
