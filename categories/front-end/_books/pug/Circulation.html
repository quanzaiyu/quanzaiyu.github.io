# 循环结构

# for

-var arr=['a', 'b', 'c']
-for (var i = 0; i < arr.length; i++)
  div=arr[i]

# for-in

pug

-var array=[100,200,300,400]
-for (var k in array)
  p=array[k] + k

html

<p>1000</p>
<p>2001</p>
<p>3002</p>
<p>4003</p>

# for-of

pug

-var array=[1,2,3,4]
-for (var v of array)
  p=v

html

<p>1</p>
<p>2</p>
<p>3</p>
<p>4</p>

# each

pug

each val in [1, 2, 3, 4, 5]
  li= val

each val, key in [1, 2, 3, 4, 5]
  li= `${key} - ${val}`

html

<li>1</li>
<li>2</li>
<li>3</li>
<li>4</li>
<li>5</li>
<li>0 - 1</li>
<li>1 - 2</li>
<li>2 - 3</li>
<li>3 - 4</li>
<li>4 - 5</li>

# 对象迭代

pug

ul
  each val, index in {1:'一',2:'二',3:'三'}
    li= index + ': ' + val

html

<ul>
  <li>1: 一</li>
  <li>2: 二</li>
  <li>3: 三</li>
</ul>

# each 的空判断

方法一

pug

- var values = []
ul
  each val in values.length ? values : ['没有内容']
    li= val

html

<ul>
  <li>没有内容</li>
</ul>

方法二: each-else

pug

- var values = []
ul
  each val in values
    li= val
  else
    li 没有内容

html

<ul>
  <li>没有内容</li>
</ul>

# while

pug

- var n = 0;
ul
  while n < 4
    li= n++

html

<ul>
  <li>0</li>
  <li>1</li>
  <li>2</li>
  <li>3</li>
</ul>
