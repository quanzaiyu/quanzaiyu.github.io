# 删除文件

需要使用到参见 del

yarn add del

使用的时候:

const gulp = require('gulp')
const del = require('del')

gulp.task('output', (cb) => {
  del([
    // 需要删掉的文件
    'dist/res/**/*',
    // 我们不希望删掉这个文件，所以我们取反这个匹配模式
    '!dist/pages/**/*',
    '!dist/libs/*'
  ], cb)
})

参考: delete-files-folder
