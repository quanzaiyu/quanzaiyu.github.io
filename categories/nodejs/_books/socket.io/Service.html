# 创建服务

# 使用 http 创建

服务器端(app.js):

var app = require('http').createServer(handler)
var io = require('socket.io')(app);
var fs = require('fs');

app.listen(80);

function handler (req, res) {
  fs.readFile(__dirname + '/index.html',
  function (err, data) {
    if (err) {
      res.writeHead(500);
      return res.end('Error loading index.html');
    }

    res.writeHead(200);
    res.end(data);
  });
}

io.on('connection', function (socket) {
  socket.emit('news', { hello: 'world' });
  socket.on('my other event', function (data) {
    console.log(data);
  });
});

socket.emit 和 socket.on：这个会worker的应该就能猜出来了，这就是一个注册事件的api，实际上，我们通过socket.io进行通讯主要就是操作各种事件，这里注册了一个叫 news 和 my other event 的事件，服务器可以触发来实现客户端与服务器的交互。

客户端(index.html):

<script src="/socket.io/socket.io.js"></script>
<script>
  var socket = io('http://localhost');
  socket.on('news', function (data) {
    console.log(data);
    socket.emit('my other event', { my: 'data' });
  });
</script>

其中 src="/socket.io/socket.io.js" 是固定写法, 实际请求的路径是 http://127.0.0.1/socket.io/socket.io.js

# 使用 Express 创建

服务器端:

const express = require('express')
const app = express();
const http = require('http').Server(app);
const io = require('socket.io')(http);

app.use(express.static('./public'));

io.on('connection', (socket) => {

});

http.listen(3000, function(){
  console.log('listening on *:3000');
});

固定的格式, socket.io是tcp连接, 传入http实例即可创建, public文件夹用于存放前端静态资源

客户端:

<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <title>socket.io</title>
</head>
<body>

<script src="/socket.io/socket.io.js"></script>
<script>
  let socket = io.connect('http://localhost');
  socket.on('news', function (data) {
    socket.emit('msg', { my: 'data' });
  });
</script>

</body>
</html>

io.connect: 这里面的路径实际上代表的是命名空间，这里是默认的命名空间 /，如果URL是 http://localhost/abc 那就代表http请求连接到localhost下的abc命名空间中

现在, 就可以愉快地编写相关的业务逻辑了

# 事件监听

服务端事件

  • connection 客户端成功连接到服务器
  • message 接收到客户端发送的消息
  • disconnect 客户端断开连接
  • error 监听错误

客户端事件

  • connect 成功连接到服务器
  • message 接收到服务器发送的消息
  • disconnect 客户端断开连接
  • error 监听错误
