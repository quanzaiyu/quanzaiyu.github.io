# 命名空间

可以把服务划分成多个命名空间，默认/不同空间不能通信

可以把服务分成多个命名空间，默认/,不同空间内不能通信

io.on('connection', socket => {});
io.of('/news').on('connection', socket => {});

客户端连接命名空间

let socket = io('http://localhost/');
let socket = io('http://localhost/news');

# 实例: 分组聊天

服务器端:

const express = require('express')
const app = express();
const http = require('http').Server(app);
const io = require('socket.io')(http);

app.use(express.static('./public'));

let group1 = io.of('/group1');
let group2 = io.of('/group2');

group1.on('connection', (socket) => {
  listening(socket, '/group1')
});

group2.on('connection', (socket) => {
  listening(socket, '/group2')
});

function listening(socket, ns) {
  socket.on('message', msg => {
    console.log(`新用户: ${socket.id}`)
    io.of(ns).emit('new user', `有新伙伴${socket.id}加入, 大家欢迎`);
  })
  socket.on('chat message', function(msg){
    console.log(`${socket.id} : ${msg}`)
    io.of(ns).emit('chat message', `${socket.id} 说: ${msg}`);
  });
  socket.on('disconnect', function () {
    console.log('断开连接')
  })
  socket.on('error', () => {
    console.log('连接错误')
  })
}

http.listen(3000, function(){
  console.log('listening on *:3000');
});

这里, 创建了两个命名空间, 并实现各自的事件监听

创建两个文件 01.html 和 02.html

# 房间

可以把一个命名空间分成多个房间，一个客户端可以同时进入多个房间。

如果大大厅里广播，那么所有在大厅里的客户端和任何房间内的客户端都能收到消息。

所有在房间里的广播和通信都不会影响到房间以外的客户端

进入房间

socket.join('chat'); // 进入chat房间

离开房间

socket.leave('chat'); // 离开chat房间

# 实例: 加入房间与离开房间

服务器端:

const express = require('express')
const app = express();
const http = require('http').Server(app);
const io = require('socket.io')(http);

app.use(express.static('./public'));

io.on('connection', (socket) => {
  socket.on('message', msg => {
    console.log(`新用户: ${socket.id}`)
  })

  socket.on('chat message', function(msg){
    console.log(`${socket.id}: ${msg}`)
    Object.keys(socket.rooms).forEach(room => {
      socket.to(room).emit('chat message', `${socket.id}: ${msg}`);
    })
  });

  socket.on('join', room => {
    console.log(`用户: ${socket.id} 加入了 ${room}`)
    socket.join(room, () => {
      socket.to(room).emit('new user', `用户: ${socket.id} 加入了 ${room}`);
    })
  })

  socket.on('leave', room => {
    console.log(`用户: ${socket.id} 离开了 ${room}`)
    socket.leave(room, () => {
      socket.to(room).emit('new user', `用户: ${socket.id} 离开了 ${room}`);
    })
  })

  socket.on('disconnect', function () {
    console.log('断开连接')
  })
  socket.on('error', () => {
    console.log('连接错误')
  })
});

http.listen(3000, function(){
  console.log('listening on *:3000');
});

其中 socket.to 会向除自己之外的房间内用户广播, 如果要包括自己请使用 io.in
