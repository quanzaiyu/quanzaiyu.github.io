# 安装与使用

安装

$ yarn add express@4.15.2 socket.io

# 搭建简易的服务器

这里使用 express 搭建一个简易的聊天室

这里创建两个文件，index.js为服务器入口文件，index.html为客户端入口文件。

index.js

var app = require('express')();
var http = require('http').Server(app);
var io = require('socket.io')(http);

app.get('/', function(req, res){
  res.sendFile(__dirname + '/index.html');
});

// 监听客户端连接
io.on('connection', function(socket){
  console.log('a user connected');
  // 监听客户端通过 send() 发送的消息, 同 emit('message')
  socket.on('message', msg => {
    console.log('客户端等候扫服务器的消息', msg)
    // 像客户端发送数据
    socket.send(`服务器回应：${msg}`)
  })
  // 监听客户端通过 emit('chat message') 发送的消息
  socket.on('chat message', function(msg){
    console.log('chat message: ' + msg);
    // 像客户端发送数据
    socket.send(`服务器说：您说的是 ${msg}`)
  });
  // 断开连接
  socket.on('disconnect', function () {
    console.log('断开连接')
  })
  // 连接错误
  socket.on('error', () => {
    console.log('连接错误')
  })
});

http.listen(3000, function(){
  console.log('listening on *:3000');
});

所有的socket处理逻辑都应该在 io.on('connection', socket => {...}) 中完成, 类似于浏览器中的 window.onload

# 广播/群发

其实只需要在服务器端发送消息, 客户端进行相应的监听即可

在上面的内容基础上加入以下内容:

index.js中添加

socket.on('chat message', function(msg){
  io.emit('chat message', msg);
});

这里，每次接收到客户端发送的消息，再将消息传递到客户端。

index.html中添加

$(function () {
  var socket = io();

  // 监听服务器 io.emit('chat message')
  socket.on('chat message', function(msg){
    $('#messages').append($('<li>').text(msg));
  });
});

首先，提交表单按钮按下后，将消息发送到服务器端，客户端同时进行监听，由于服务器又将消息传递出来，客户端显示此消息。

至此, 一个简单的聊天室完成, 但是这只是基础功能, 没有用户认证等等。

# 在Vue中使用

在ES6中需要安装 socket.io-client:

$ yarn add socket.io-client @types/socket.io-client

在Vue组件中, 我是用ts写的, 普通js同理

import { Vue, Component } from 'vue-property-decorator'

import io from 'socket.io-client'
const socket = io('http://localhost:3000')

@Component
export default class App extends Vue {
  async created() {
    socket.on('connect', function() {
      console.log('连接成功')
      // 客户端连接成功后向服务器发送消息 'welcome'
      socket.send('welcome')
    })
    // 客户端收到服务器发过来的消息后触发
    socket.on('message', function(message: any) {
      console.log(message)
    })
    // 监听与服务器端断开连接事件
    socket.on('disconnect', function() {
      console.log('断开连接')
    })
  }
}
