# 安装

$ yarn add sequelize

# 还需要安装以下之一：
$ yarn add pg pg-hstore  // postgreSql
$ yarn add mysql // mysql 或 mariadb
$ yarn add sqlite3 // sqlite
$ yarn add tedious // MSSQL
