# 定义模型

# 通过 init 定义

const Sequelize = require('sequelize');
const sequelize = new Sequelize('mysql://root:root@localhost:3306/test');

class Article extends Sequelize.Model {}
Article.init({
  id:{
    type: Sequelize.INTEGER,
    primaryKey: true,
    allowNull: true,
    autoIncrement: true
  },
  title: Sequelize.STRING,
  author: Sequelize.STRING,
  content: Sequelize.STRING,
  category: Sequelize.STRING,
  createdAt: Sequelize.DATE,
  updatedAt: Sequelize.DATE
}, {
  sequelize,
  freezeTableName: true
})

sequelize.sync({force: false}) // 自动创建表

class ArticleModel extends Article { // 继承以使用模型方法
  static async createArticle(data){
    return await Article.create({
      title: data.title,
      author: data.author,
      content: data.content,
      category: data.category
    });
  }

  static async getArticleDetail(id){
    return await Article.findOne({
      where:{
        id
      }
    });
  }
}

module.exports = ArticleModel;

# 通过 sequelize.define 定义

格式:

sequelize.define(modelName, attributes, [options]) -> Model

这个实例方法用于定义一个新Model（模型）。Model相当于数据库中的表，可以通过 sequelize.define() 或 sequelize.import() 方法创建。

表中的字段通过第二个参数对象attributes来定义，对象中的一个属性相当于表中的一个字段。

如，可以像下面这样定义一个表：

const Sequelize = require('sequelize');
const sequelize = new Sequelize('mysql://root:root@localhost:3306/test');

const Article = sequelize.define('article', {
  id: {
    type: Sequelize.INTEGER,
    primaryKey: true,
    allowNull: true,
    autoIncrement: true
  },
  title: Sequelize.STRING,
  author: {
    type: Sequelize.STRING,
    validate: {
      is: ["[a-z]",'i'],        // will only allow letters
      max: 23,                  // only allow values <= 23
      isIn: {
        args: [['en', 'zh']],
        msg: "Must be English or Chinese"
      }
    },
    field: 'author'
    // Other attributes here
  },
  content: Sequelize.STRING,
  category: Sequelize.STRING,
  createdAt: Sequelize.DATE,
  updatedAt: Sequelize.DATE
}, {
  freezeTableName: true
});

sequelize.sync({force: false}) // 自动创建表

class ArticleModel extends Article { // 继承以使用模型方法
  static async createArticle(data){
    return await Article.create({
      title: data.title,
      author: data.author,
      content: data.content,
      category: data.category
    });
  }

  static async getArticleDetail(id){
    return await Article.findOne({
      where:{
        id
      }
    });
  }
}

module.exports = ArticleModel;

# 通过模型导入的方式

格式:

sequelize.import(path) -> Model

schema/article.js

module.exports = function(sequelize, DataTypes){
  return sequelize.define('article', {
    id: {
      type: DataTypes.INTEGER,
      primaryKey: true,
      allowNull: true,
      autoIncrement: true
    },
    title: DataTypes.STRING,
    author: DataTypes.STRING,
    content: DataTypes.STRING,
    category: DataTypes.STRING,
    createdAt: DataTypes.DATE,
    updatedAt: DataTypes.DATE
  }, {
    freezeTableName: true
  });
}

model/article.js

const sequelize = new Sequelize('mysql://root:root@localhost:3306/test');

// 引入数据表模型
const Article = sequelize.import('../schema/article');
Article.sync({force: false}); // 自动创建表

class ArticleModel extends Article { // 继承以使用模型方法
  static async createArticle(data){
    return await Article.create({
      title: data.title,
      author: data.author
    });
  }

  static async getArticleDetail(id){
    return await Article.findOne({
      where:{
        id
      }
    });
  }
}

module.exports = ArticleModel;
