# Scrapy: 基础使用

# URL 的两种写法

定义爬虫其实路径有两种方式:

  1. 重写 start_requests(self) 方法并手动解析
  2. 定义 start_urls 类常量以自动解析, 并且重写解析方法 parse()

不过无论哪一种写法，URL都是必须的

# start_requests

  1. 定义链接；
  2. 通过链接爬取(下载)页面；
  3. 定义规则，然后提取数据 (以下示例只是简单的保存, 并未做数据解析)；
import scrapy

class mingyan(scrapy.Spider): #需要继承scrapy.Spider类
    name = "mingyan" # 定义蜘蛛名
    def start_requests(self): # 由此方法通过下面链接爬取页面
        # 定义爬取的链接
        urls = [
            'http://lab.scrapyd.cn/page/1/',
            'http://lab.scrapyd.cn/page/2/',
        ]
        for url in urls:
            yield scrapy.Request(url=url, callback=self.parse) #爬取到的页面如何处理？提交给parse方法处理(parse名字可自定义)
    def parse(self, response):
        page = response.url.split("/")[-2]     #根据上面的链接提取分页,如：/page/1/，提取到的就是：1
        filename = 'mingyan-%s.html' % page    #拼接文件名，如果是第一页，最终文件名便是：mingyan-1.html
        with open(filename, 'wb') as f:        #python文件操作，不多说了；
            f.write(response.body)             #刚才下载的页面去哪里了？response.body就代表了刚才下载的页面！
        self.log('保存文件: %s' % filename)     # 打个日志

执行 scrapy crawl mingyan 进行爬取

# start_urls

"""
    scrapy初始Url的两种写法，
    一种是常量start_urls，并且需要定义一个方法parse（）
    另一种是直接定义一个方法：star_requests()
"""
import scrapy
class simpleUrl(scrapy.Spider):
    name = "simpleUrl"
    start_urls = [  #另外一种写法，无需定义start_requests方法
        'http://lab.scrapyd.cn/page/1/',
        'http://lab.scrapyd.cn/page/2/',
    ]
    def parse(self, response):
        page = response.url.split("/")[-2]
        filename = 'mingyan-%s.html' % page
        with open(filename, 'wb') as f:
            f.write(response.body)
        self.log('保存文件: %s' % filename)

执行 scrapy crawl simpleUrl 进行爬取

# 提取内容

使用 extract() 方法提取获取到的内容, 返回的是一个 list, 如果只想获取其匹配的第一条数据, 使用 extract_first()。比如:

text = v.css('.text::text').extract_first()  # 提取名言
autor = v.css('.author::text').extract_first()  # 提取作者
tags = v.css('.tags .tag::text').extract()  # 提取标签(列表)

以下两种写法效果一样:

response.css('title').extract()[0]
response.css('title').extract_first()

提取 .post-content 内的所有内容:

response.css(".post-content *::text").extract()
