# 数据库: 增删改查

# 插入操作

使用 insert 语句可以向表中添加一条数据，如果返回为 1 则添加成功。

public function insert()
{
  $data = [
    'name' => 'qzy',
    'password' => 555555
  ];
  return Db::table('user')
    ->insert($data) ? '添加成功' : '添加失败';
}

转换为 SQL 为:

INSERT INTO `user` (`name` , `password`) VALUES ('qzy' , '555555')

# replace

第二参数为 replace，只有数据库为 MySQL 的时候才能传入 true:

$data = [
  'id' => 4,
  'name' => 'qzy',
  'password' => 555555
];
return Db::table('user')
  ->insert($data, true) ? '添加成功' : '添加失败';

转换为 SQL 为:

REPLACE INTO `user` (`id`,`name` , `password`) VALUES (4, 'qzy' , '555555')

# 使用 data 包装

$data = [
  'name' => 'qzy',
  'password' => 555555
];
return Db::table('user')
  ->data($data)
  ->insert() ? '添加成功' : '添加失败';

转换为 SQL 为:

INSERT INTO `user` (`name` , `password`) VALUES ('qzy' , '555555')

# 插入的同时返回新增主键 id

使用 insertGetId 可以在插入的同时返回新增主键 id

$data = [
  'name' => 'qzy',
  'password' => 555555
];
return Db::table('user')
  ->insertGetId($data);

# 多条插入

使用 insertAll 插入多条数据，返回插入的条数。

public function insertAll()
{
  $datas = [[
      'name' => 'qzy2',
      'password' => 555555
    ],[
      'name' => 'qzy3',
      'password' => 555555
    ],[
      'name' => 'qzy4',
      'password' => 555555
    ]
  ];
  return Db::table('user')
    ->insertAll($datas);
}

转换为 SQL 为:

INSERT INTO `user` (`name` , `password`) VALUES ( 'qzy2','555555' ) , ( 'qzy3','555555' ) , ( 'qzy4','555555' )

insertAll 不支持 data 包装。

# 更新数据

使用 update 语句可以更新表中的数据，如果返回为 1 则更新成功。

更新之前必须带有 where 更新条件。

public function update()
{
    return Db::table('user')
      ->where('id', 12)
      ->update(['name' => '小昱']);
}

如果更新条件为主键，则可以直接把主键写在更新数组中

public function update()
{
    return Db::table('user')
      ->update(['name' => '小昱', 'id' => 12]);
}

转换为 SQL 为:

UPDATE `user` SET `name` = '小昱' WHERE `id` = 12

# 删除数据

使用 delete 语句进行数据删除操作，如果删除成功，返回 1 。

public function delete()
{
    return Db::table('user')
      ->where('id', 22)
      ->delete();
}

如果更新条件为主键，则可以直接把主键写在删除参数中

public function delete()
{
    return Db::table('user')
      ->delete(20);
}

转换为 SQL 为:

 DELETE FROM `user` WHERE `id` = 22

# 原生查询

使用 Db::query 可以直接使用原生查询操作

public function query()
{
    $sql = 'SELECT * from `user` where id=12';
    $res = Db::query($sql);
    dump($res);
}

# 原生写操作: 更新、删除、添加

使用 Db::execute 可以进行原生写操作

比如:

public function execute()
{
    $sql = 'UPDATE `user` SET `name` = \'小昱live\' WHERE `id` = 12';
    return Db::execute($sql);
}

如果执行成功，则返回 1。
