# 响应

# 引入响应对象的几种方法

方法一: 使用 Facade 静态代理

public function index($name) {
  return \think\facade\Response::code(200)
    ->data('Hello, ' . $name)
    ->contentType('text/plain');
}

方法二: 使用实例化 Response 类

public function index($name) {
  $response = new Response();
  return $response
    ->data('Hello, ' . $name)
    ->code(200)
    ->contentType('text/plain');
}

方法三: 使用依赖注入

public function index(\think\Response $response, $name) {
  return $response
    ->data('Hello, ' . $name)
    ->code(200)
    ->contentType('text/plain');
}

方法四: 使用 Controller 的 response 属性

public function index($name) {
  return $this->response
      ->data('Hello, ' . $name)
      ->code(200)
      ->contentType('text/plain');
}

方法五: 使用助手函数

public function index($name) {
  return response()
    ->data('Hello,' . $name)
    ->code(200)
    ->contentType('text/plain');
}

# 设置状态码

使用 response 对象的 code 方法可以设置状态码:

Response::code(401);

更多时候，会使用助手函数返回状态码:

json($data, 201);
view($data, 401);

相当于

json($data)->code(201);

# 设置数据

使用 response 对象的 data 方法可以设置数据:

response()->data($data);
json()->data($data);

最终的输出数据就是 json_encode($data) 转换后的数据。

# 重定向

可以使用 redirect 助手函数进行重定向

public function hello() {
  return redirect('http://xiaoyulive.top');
}

# HTTP 响应头

可以使用静态代理 \think\facade\Response::header() 设置 HTTP 响应头信息 :

比如设置一个自定义响应头 :

public function test () {
  return \think\facade\Response::header('token', 'xxxxxx');
}

# 跨域设置

前端默认会有同源策略，进行跨域拦截

通常解决方案为 跨域资源共享（CORS）

public function test () {
  return \think\facade\Response::header([
    'Access-Control-Allow-Credentials' => 'true', // 后端允许发送 Cookie
    'Access-Control-Allow-Origin' => 'http://192.168.1.22:8080',  // 允许访问的域（协议+域名+端口）
    'Access-Control-Allow-Methods' => 'GET, POST, OPTIONS', // 允许通过的请求方式
    'Access-Control-Allow-Headers' => 'X-PINGOTHER, Content-Type' // 表明服务器允许请求中携带字段 X-PINGOTHER 与 Content-Type
  ]);
}
