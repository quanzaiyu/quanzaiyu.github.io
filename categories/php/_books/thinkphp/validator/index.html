# 验证器

ThinkPHP 中自带验证器模块，只需要在自定义验证器中继承即可，核心文件位于 thinkphp/library/think/Validate.php 中。

一般，我们会在 application 下新建一个 validate 文件夹，用于存放自定义验证文件。

比如创建一个验证个人信息的验证器 :

application/validate/User.php

namespace app\validate;
use think\Validate;

class User extends Validate
{
  protected $rule = [
    'name|姓名' => [
      'require',
      'min' => 5,
      'max' => 20
    ],
    'email|邮箱' => [
      'require',
      'email'
    ],
    'password|密码' => [
      'require',
      'min' => 3,
      'max' => 12,
      'alphaNum'
    ],
    'mobile|电话' => [
      'require',
      'mobile'
    ]
  ];
}

可以看到，这里只需要继承 think\Validate 即可轻松实现验证器，通过重写 $rule 属性创建自定义验证规则。

$rule 中指定要验证的字段名，通过 | 对字段进行命名，如果不命名默认为字段名，在输出提示中会显示。

要使用这个验证器，在控制器中指定 :

application/index/controller/Index.php

namespace app\index\controller;

use think\Controller;
use app\validate\User;

class Index extends Controller
{
  public function index() {
    $validate = new User();
    $data = [
      'name' => 'xiaoyu',
      'email' => 'xiaoyu@xiaoyulive.top',
      'password' => 123456,
      'mobile' => 18388019780
    ];
    if(!$validate->check($data)) {
      return $validate->getError();
    }
    return '验证成功';
  }
}

可以看到，首先需要 new 一个自定义验证器，通过验证器的 check 方法验证数据，如果验证失败使用 getError 获取错误信息。

也可使用助手函数 validate()，比如

if(!validate('User')->check($_GET)) {
  return validate('User')->getError();
}

或者 Loader:

$validate = Loader::validate('User');

# 验证格式

验证器可以是一个数组，也可以使用字符串，比如:

$rule = [
  'name' => [
    'require',
    'min' => 5,
    'max' => 20
  ],
  'age' => [
    'between' => '0, 120'
  ]
]

使用字符串可以写为 :

$rule = [
  'name' => 'require|min:5|max:20',
  'age' => 'between:0,120'
]

每个验证规则之间使用 | 分割。

# 静态代理验证器

使用 Facade 可以很容易的对自定义验证器进行静态代理。

application/facade/User.php

namespace app\facade;
use think\Facade;

class User extends Facade
{
  protected static function getFacadeClass()
  {
    return 'app\validate\User';
  }
}

application/index/controller/Index.php

namespace app\index\controller;

use think\Controller;
use app\facade\User;

class Index extends Controller
{
  public function validateTest() {
    $data = [
      'name' => 'xiaoyu',
      'email' => 'xiaoyu@xiaoyulive.top',
      'password' => 123456,
      'mobile' => 18388019782
    ];
    if(!User::check($data)) {
      return User::getError();
    }
    return '验证成功';
  }
}

这里，app\validate\User 跟上一小节的一样，只是使用静态代理辅助使用验证器。

# 调用控制器中的验证器

namespace app\index\controller;

use think\Controller;

class Index extends Controller
{
  public function validateWithController() {
    // 要验证的数据
    $data = [
      'name' => 'xiaoyu',
      'email' => 'xiaoyu@xiaoyulive.top',
      'password' => 123456,
      'mobile' => 18388019782
    ];

    // 验证规则
    $validate = 'app\validate\User';

    $res = $this -> validate($data, $validate);

    if(true !== $res) {
      return $res;
    }
    return '验证成功';
  }
}

在控制器中可以直接调用 $this -> validate ，传入要验证的数据和规则即可，规则可以引用之前创建的自定义验证器类。

注意到，可以传入两个参数:

第一个参数是要验证的数据，第二参数是要使用的验证器。

# 使用控制器进行独立验证

如果是独立验证，在控制器中跟使用验证器的方法一样。

$rules = [
  ['name|姓名','require|min: 1|max: 5','名称必须|名字不能为空|名字最多不能超过5个字符'],
  ['email|邮箱','require|email','邮箱必须|邮箱格式不正确'],
];
$res = $this->validate($_GET, $rules);
if(true !== $res) {
  return $res;
}

# 模型验证

在控制器中实例化模型，可以直接使用验证器中的方法:

$User = new User;
$result = $User->validate(
  [
    'name'  => 'require|max:25',
    'email' => 'email',
  ],
  [
    'name.require' => '名称必须',
    'name.max'     => '名称最多不能超过25个字符',
    'email'        => '邮箱格式错误',
  ]
)->save($data);
if(false === $result){
  // 验证失败 输出错误信息
  dump($User->getError());
}

注意到，getError 是模型实例下的方法。

# 独立验证

使用验证器 think\Validate 中的 rule() 方法添加验证规则，不需要创建单独的验证器类。

这里使用 facade 类包装的 Validate 类，当然也可以使用原始的 Validate。

namespace app\index\controller;

use think\Controller;
use think\facade\Validate;

class Index extends Controller
{
  public function validateIndependent() {
    // 需要校验的数据
    $data = [
      'name' => 'xiaoyu',
      'email' => 'xiaoyu@xiaoyulive.top',
      'password' => 123456,
      'mobile' => 18388019782
    ];
    // 校验规则
    $rule = [
      'name|姓名' => 'require|min: 5|max: 20',
      'email|邮箱' => 'require|email',
      'password|密码' => 'require|min: 3|max: 12|alphaNum',
      'mobile|电话' => 'require|mobile'
    ];
    // 添加字段的验证规则
    Validate::rule($rule);
    // 绑定校验数据
    $res = Validate::check($data);

    if(!$res) {
      return Validate::getError();
    }
    return '验证成功';
  }
}

也可在实例化验证器的时候传入验证规则:

$rules = [
  'name|姓名' => 'require|min: 1|max: 5',
  'email|邮箱' => 'require|email'
];
$validate = new \think\Validate($rules);
if(!$validate->check($_GET)) {
  return $validate->getError();
}

注意

不能使用助手函数直接传入验证规则实例化:

$validate = validate($rules);

# 闭包验证

$validate = new \think\Validate([
  'name'  => function($value,$data) { 
    return 'thinkphp' == $value;
  },
]);

# 自定义验证提示信息

在验证器中，ThinkPHP 自身有封装的提示信息，当验证失败，使用 getError() 方法会返回这些错误信息。

有的时候，系统自身的提示信息不能满足我们的需求，我们可以自定义提示信息，通过重写验证器的 $message 指定。

namespace app\index\validate;

use think\Validate;

class User extends Validate
{
  protected $rule =   [
    'name'  => 'require|max:25',
    'age'   => 'number|between:1,120',
    'email' => 'email',
  ];

  protected $message  =   [
    'name.require' => '名称必须',
    'name.max'     => '名称最多不能超过25个字符',
    'age.number'   => '年龄必须是数字',
    'age.between'  => '年龄只能在1-120之间',
    'email'        => '邮箱格式错误',
  ];
}

如果是独立验证，使用 message() 方法绑定自定义提示信息 :

$message = [
  'name.require' => '请输入姓名'
];
Validate::message($message);

也可以在初始化的时候同时传入提示信息:

$validate = new \think\Validate($rule, $msg);

# 设置字段信息

除了在验证规则中直接设置字段信息之外，还支持传入 field 参数批量设置字段的描述信息，例如：

$rule = [
  'name'  => 'require|max:25',
  'age'   => 'number|between:1,120',
  'email' => 'email',
];

$field = [
  'name'  => '名称',
  'age'   => '年龄',
  'email' => '邮箱',
];

$data = [
  'name'  => 'thinkphp',
  'age'   => 10,
  'email' => 'thinkphp@qq.com',
];

$validate = new Validate($rule, [] , $field);
$result   = $validate->check($data);

# 所有验证信息一起设置

可以将所有验证信息一起设置，包括字段信息、验证规则、提示信息:

$rules = [
  ['name|姓名','require|min: 1|max: 5','名称必须|名字不能为空|名字最多不能超过5个字符'],
  ['email|邮箱','require|email','邮箱必须|邮箱格式不正确'],
];
$validate = new \think\Validate($rules);
if(!$validate->check($_GET)) {
  return $validate->getError();
}

# 验证场景

可以在定义验证规则的时候定义场景，并且验证不同场景的数据，例如：

$rule = [
  'name'  => 'require|max:25',
  'age'   => 'number|between:1,120',
  'email' => 'email',
];
$msg = [
  'name.require' => '名称必须',
  'name.max'     => '名称最多不能超过25个字符',
  'age.number'   => '年龄必须是数字',
  'age.between'  => '年龄只能在1-120之间',
  'email'        => '邮箱格式错误',
];
$data = [
  'name'  => 'thinkphp',
  'age'   => 10,
  'email' => 'thinkphp@qq.com',
];
$validate = new Validate($rule);
$validate->scene('edit', ['name', 'age']);
$result = $validate->scene('edit')->check($data);

表示验证edit场景（该场景定义只需要验证name和age字段）。

如果使用了验证器，可以直接在类里面定义场景，例如：

namespace app\index\validate;

use think\Validate;

class User extends Validate
{
  protected $rule =   [
    'name'  => 'require|max:25',
    'age'   => 'number|between:1,120',
    'email' => 'email',    
  ];
  protected $message  =   [
    'name.require' => '名称必须',
    'name.max'     => '名称最多不能超过25个字符',
    'age.number'   => '年龄必须是数字',
    'age.between'  => '年龄只能在1-120之间',
    'email'        => '邮箱格式错误',    
  ];
  protected $scene = [
    'edit'  =>  ['name','age'],
  ];
}

然后再需要验证的地方直接使用 scene 方法验证

$data = [
  'name'  => 'thinkphp',
  'age'   => 10,
  'email' => 'thinkphp@qq.com',
];

$validate = new \app\index\validate\User($rule);
$result = $validate->scene('edit')->check($data);

可以在定义场景的时候对某些字段的规则重新设置，例如：

namespace app\index\validate;

use think\Validate;

class User extends Validate
{
  protected $rule =   [
    'name'  => 'require|max:25',
    'age'   => 'number|between:1,120',
    'email' => 'email',    
  ];
  protected $message  =   [
    'name.require' => '名称必须',
    'name.max'     => '名称最多不能超过25个字符',
    'age.number'   => '年龄必须是数字',
    'age.between'  => '年龄只能在1-120之间',
    'email'        => '邮箱格式错误',    
  ];
  protected $scene = [
    'edit'  =>  ['name','age'=>'require|number|between:1,120'],
  ];
}

可以对场景设置一个回调方法，用于动态设置要验证的字段，例如：

$rule = [
  'name'  => 'require|max:25',
  'age'   => 'number|between:1,120',
  'email' => 'email',
];
$msg = [
  'name.require' => '名称必须',
  'name.max'     => '名称最多不能超过25个字符',
  'age.number'   => '年龄必须是数字',
  'age.between'  => '年龄只能在1-120之间',
  'email'        => '邮箱格式错误',
];
$data = [
  'name'  => 'thinkphp',
  'age'   => 10,
  'email' => 'thinkphp@qq.com',
];
$validate = new Validate($rule);
$validate->scene('edit', function($key,$data){
  return 'email'==$key && isset($data['id'])? true : false;
});
$result = $validate->scene('edit')->check($data);
