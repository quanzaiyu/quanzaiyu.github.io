# 与 Mybatis Plus 融合

最近在学习 Mybatis Plus 的使用，希望通过 Spring Boot 快速将 Mybatis Plus 整合进来。

对于 Spring Boot 项目，Mybatis Plus 团队也有自己的启动器: mybatis-plus-boot-starter。这个依赖内部已经整合了 mybatis-spring，也包括非快速启动的mybatis-plus（这个依赖需要额外的配置数据源等信息），所以如果您在网上看到引入两个和 mybatis-plus 有关的依赖的话完全是多余的。

# 项目结构

创建如下项目结构:

src
└─main
├─java
│  └─com.example.demo
│     ├─config     # 配置
│     ├─controller # 控制器
│     │   └─TestController.java
│     ├─dao        # 数据访问对象 Data Access Object
│     ├─dto        # 数据传输对象 Data Transfer Object
│     ├─entity     # 数据库实体类
│     ├─pojo       # 自建实体类 Plain Ordinary Java Object
│     ├─service    # 服务层
│     │   └─Impl   # 实现
│     └─Application.java
└─resources
└─application.yaml

# 配置项目依赖

首先在 pom.xml 中引入 mybatis-plus-boot-starter 和 mysql-connector-java:

<!-- spring-boot整合mybatis-plus -->
<dependency>
  <groupId>com.baomidou</groupId>
  <artifactId>mybatis-plus-boot-starter</artifactId>
  <version>3.0.6</version>
</dependency>
<!-- mysql驱动 -->
<dependency>
  <groupId>mysql</groupId>
  <artifactId>mysql-connector-java</artifactId>
  <version>8.0.15</version>
</dependency>

# 配置数据源

再修改 application.yaml, 配置数据源

server:
  port: 8088 #端口配置
spring:
  application:
    name: test #实例名
  #配置数据源
  datasource:
    driver-class-name: com.mysql.cj.jdbc.Driver
    url: jdbc:mysql://localhost:3306/award?useUnicode=true&characterEncoding=utf8&serverTimezone=UTC
    username: root
    password: root

Spring Boot 会自动加载 spring.datasource.* 相关配置，数据源就会自动注入到 sqlSessionFactory 中，sqlSessionFactory 会自动注入到 Mapper 中

# 配置 Mybatis Plus

在 application.yaml 中添加 mybatis-plus 配置选项

mybatis-plus:
  global-config:
    db-config:
      db-column-underline: true
  configuration:
    # 开启下划线到驼峰的自动转换. 作用：将数据库字段根据驼峰规则自动注入到对象属性。
    map-underscore-to-camel-case: true

# 配置自动扫描

在配置中添加对 mapper 包扫描 @MapperScan

com.example.demo.config/MybatisPlusConfig.java

package com.example.demo.config;

import com.baomidou.mybatisplus.extension.plugins.PaginationInterceptor;

import org.mybatis.spring.annotation.MapperScan;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.transaction.annotation.EnableTransactionManagement;

@EnableTransactionManagement
@Configuration
@MapperScan("com.example.demo.dao") // 扫描 dao 包
public class MybatisPlusConfig {
  /**
    * 分页插件
    */
  @Bean
  public PaginationInterceptor paginationInterceptor() {
    return new PaginationInterceptor();
  }
}

或者直接在 Mapper 类上面添加注解 @Mapper，如果配置 @MapperScan 可以省略 @Mapper

如果不用插件, 可以直接在 Application.java 中配置:

@SpringBootApplication
@EnableScheduling
@MapperScan("com.example.demo.dao") // 扫描 mapper
public class Application {
	public static void main(String[] args) {
		SpringApplication.run(Application.class, args);
	}
}

# 创建实体类

com.example.demo.entity/User.java

package com.example.demo.entity;

import com.baomidou.mybatisplus.annotation.IdType;
import com.baomidou.mybatisplus.annotation.TableId;
import com.baomidou.mybatisplus.annotation.TableName;
import com.baomidou.mybatisplus.extension.activerecord.Model;

import lombok.AllArgsConstructor;
import lombok.Data;
import lombok.NoArgsConstructor;

@Data
@AllArgsConstructor
@NoArgsConstructor
@TableName("user")
public class User extends Model<User>{
  @TableId(type = IdType.ID_WORKER_STR)
  private String id;
  private String name;
  private String avatar;

  public User(String name, String avatar) {
      this.name = name;
      this.avatar = avatar;
  }
}

# 以实体类作为参数

可以直接以实体类作为参数接收:

package com.example.demo.controller;

import com.example.demo.config.Result;
import com.example.demo.entity.User;
import com.example.demo.service.UserService;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RestController;

import java.util.Date;

@RestController
@RequestMapping(value="/test", produces="application/json; charset=utf-8")
public class TestController {
    @Autowired
    UserService userService;

    @RequestMapping(value = "add")
    public Result add(User user){
        user.setCreateTime(new Date());
        return Result.ok(userService.save(user));
    }
}

请求时, 以普通参数传递即可:

GET: http://localhost:8088/test/add?name=yu

POST:

fetch('http://localhost:8088/test/add', {
  method: 'post',
  body: 'name=yu',
  headers: {
    'Content-Type': "application/x-www-form-urlencoded",
  }
})

# 开发 Mapper

使用 Mapper 作 SQL 生产

package com.example.demo.mapper;

import com.example.demo.entity.User;

import org.apache.ibatis.annotations.Delete;
import org.apache.ibatis.annotations.Insert;
import org.apache.ibatis.annotations.Result;
import org.apache.ibatis.annotations.Results;
import org.apache.ibatis.annotations.Select;
import org.apache.ibatis.annotations.Update;

import java.util.List;

public interface UserMapper {
    @Select("SELECT * FROM user")
    @Results({
            @Result(property = "name",  column = "name"),
            @Result(property = "avatar", column = "avatar")
    })
    List<User> getAll();

    @Select("SELECT * FROM user WHERE id = #{id}")
    @Results({
            @Result(property = "name",  column = "name"),
            @Result(property = "avatar", column = "avatar")
    })
    User getOne(Long id);

    @Insert("INSERT INTO user(name,avatar) VALUES(#{name}, #{avatar})")
    int insert(User user);

    @Update("UPDATE user SET name=#{name},avatar=#{avatar} WHERE id =#{id}")
    void update(User user);

    @Delete("DELETE FROM user WHERE id =#{id}")
    void delete(Long id);
}
  • @Select 是查询类的注解，所有的查询均使用这个
  • @Result 修饰返回的结果集，关联实体类属性和数据库字段一一对应，如果实体类属性和数据库属性名保持一致，就不需要这个属性来修饰。
  • @Insert 插入数据库使用，直接传入实体类会自动解析属性到对应的值
  • @Update 负责修改，也可以直接传入对象
  • @delete 负责删除

编写单元测试:

package com.example.demo;

import com.example.demo.entity.User;
import com.example.demo.mapper.UserMapper;

import org.junit.Assert;
import org.junit.Test;
import org.junit.runner.RunWith;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.test.context.SpringBootTest;
import org.springframework.test.context.junit4.SpringRunner;

import java.util.List;

@RunWith(SpringRunner.class)
@SpringBootTest
public class UserMapperTest {

    @Autowired
    private UserMapper userMapper;

    @Test
    public void testInsert() throws Exception {
        userMapper.insert(new User("aa1", "a123456"));
        userMapper.insert(new User("bb1", "b123456"));
        userMapper.insert(new User("cc1", "b123456"));
    }

    @Test
    public void testQuery() throws Exception {
        List<User> users = userMapper.getAll();
        System.out.println(users.toString());
    }

    @Test
    public void testUpdate() throws Exception {
        User user = userMapper.getOne(1l);
        System.out.println(user.toString());
        user.setName("quanzaiyu");
        userMapper.update(user);
        Assert.assertTrue(("quanzaiyu".equals(userMapper.getOne(1l).getName())));
    }
}
